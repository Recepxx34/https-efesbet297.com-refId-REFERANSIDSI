211    private boolean isWritableFile(File file) {
212        return !file.exists() || file.isFile() && file.canWrite();
213    }

216     * Validates the input parameters.
217     */
218    public void validate() throws PackagingException {
219        if (control == null || !control.isDirectory()) {
220            throw new PackagingException("The 'control' attribute doesn't point to a directory.");
221        }
222
223        if (changesIn != null) {
224
225            if (changesIn.exists() && (!changesIn.isFile() || !changesIn.canRead())) {
226                throw new PackagingException("The 'changesIn' setting needs to point to a readable file. " + changesIn + " was not found/readable.");
227            }
228
229            if (changesOut != null && !isWritableFile(changesOut)) {
230                throw new PackagingException("Cannot write the output for 'changesOut' to " + changesOut);
231            }
232
233            if (changesSave != null && !isWritableFile(changesSave)) {
234                throw new PackagingException("Cannot write the output for 'changesSave' to " + changesSave);
235            }
236
237        } else {
238            if (changesOut != null || changesSave != null) {
239                throw new PackagingException("The 'changesOut' or 'changesSave' settings may only be used when there is a 'changesIn' specified.");
240            }
241        }
242
243        if (Compression.toEnum(compression) == null) {
244            throw new PackagingException("The compression method '" + compression + "' is not supported (expected 'none', 'gzip', 'bzip2' or 'xz')");
245        }
246
247        if (deb == null) {
248            throw new PackagingException("You need to specify where the deb file is supposed to be created.");
249        }
250    }

252    public void makeDeb() throws PackagingException {
253        BinaryPackageControlFile packageControlFile;
254        try {
255            console.info("Creating debian package: " + deb);
256
257            // If we should sign the package
258            boolean doSign = signPackage;
259
260            if (doSign) {
261
262                if (keyring == null || !keyring.exists()) {
263                    doSign = false;
264                    console.warn("Signing requested, but no keyring supplied");
265                }
266
267                if (key == null) {
268                    doSign = false;
269                    console.warn("Signing requested, but no key supplied");
270                }
271
272                if (passphrase == null) {
273                    doSign = false;
274                    console.warn("Signing requested, but no passphrase supplied");
275                }
276
277                FileInputStream keyRingInput = new FileInputStream(keyring);
278                PGPSigner signer = null;
279                try {
280                    signer = new PGPSigner(new FileInputStream(keyring), key, passphrase);
281                } finally {
282                    keyRingInput.close();
283                }
284
285                int digest = PGPUtil.SHA1;
286
287                PGPSignatureGenerator signatureGenerator = new PGPSignatureGenerator(new BcPGPContentSignerBuilder(signer.getSecretKey().getPublicKey().getAlgorithm(), digest));
288                signatureGenerator.init(PGPSignature.BINARY_DOCUMENT, signer.getPrivateKey());
289
290                packageControlFile = createSignedDeb(Compression.toEnum(compression), signatureGenerator, signer);
291            } else {
292                packageControlFile = createDeb(Compression.toEnum(compression));
293            }
294
295        } catch (Exception e) {
296            throw new PackagingException("Failed to create debian package " + deb, e);
297        }
298
299        makeChangesFiles(packageControlFile);
300    }

302    private void makeChangesFiles(final BinaryPackageControlFile packageControlFile) throws PackagingException {
303        if (changesOut == null) {
304            changesOut = new File(deb.getParentFile(), deb.getName().replace(".deb", ".changes"));
305        }
306
307        ChangesProvider changesProvider;
308        FileOutputStream out = null;
309
310        try {
311            console.info("Creating changes file: " + changesOut);
312
313            out = new FileOutputStream(changesOut);
314
315            if (changesIn != null && changesIn.exists()) {
316                // read the changes form a textfile provider
317                changesProvider = new TextfileChangesProvider(new FileInputStream(changesIn), packageControlFile);
318            } else {
319                // create an empty changelog
320                changesProvider = new ChangesProvider() {
321                    @Override
322                    public ChangeSet[] getChangesSets() {
323                        return new ChangeSet[] {
324                                new ChangeSet(packageControlFile.get("Package"),
325                                        packageControlFile.get("Version"),
326                                        new Date(),
327                                        packageControlFile.get("Distribution"),
328                                        packageControlFile.get("Urgency"),
329                                        packageControlFile.get("Maintainer"),
330                                        new String[0])
331                        };
332                    }
333                };
334            }
335
336            ChangesFileBuilder builder = new ChangesFileBuilder();
337            ChangesFile changesFile = builder.createChanges(packageControlFile, deb, changesProvider);
338
339            if (keyring != null && key != null && passphrase != null) {
340                console.info("Signing the changes file with the key " + key);
341                PGPSigner signer = new PGPSigner(new FileInputStream(keyring), key, passphrase);
342                signer.clearSign(changesFile.toString(), out);
343            } else {
344                out.write(changesFile.toString().getBytes("UTF-8"));
345            }
346            out.flush();
347
348        } catch (Exception e) {
349            throw new PackagingException("Failed to create the Debian changes file " + changesOut, e);
350        } finally {
351            IOUtils.closeQuietly(out);
352        }
353
354        if (changesSave == null || !(changesProvider instanceof TextfileChangesProvider)) {
355            return;
356        }
357
358        try {
359            console.info("Saving changes to file: " + changesSave);
360
361            ((TextfileChangesProvider) changesProvider).save(new FileOutputStream(changesSave));
362
363        } catch (Exception e) {
364            throw new PackagingException("Failed to save debian changes file " + changesSave, e);
365        }
366    }

368    private List<String> populateConffiles(Collection<DataProducer> producers) {
369        final List<String> result = new ArrayList<String>();
370
371        if (producers == null || producers.isEmpty()) {
372            return result;
373        }
374
375        final DataConsumer receiver = new DataConsumer() {
376            public void onEachFile(InputStream input, TarArchiveEntry entry)  {
377                String tempConffileItem = entry.getName();
378                if (tempConffileItem.startsWith(".")) {
379                    tempConffileItem = tempConffileItem.substring(1);
380                }
381                console.info("Adding conffile: " + tempConffileItem);
382                result.add(tempConffileItem);
383            }
384
385            public void onEachLink(TarArchiveEntry entry)  {
386            }
387
388            public void onEachDir(String dirname, String linkname, String user, int uid, String group, int gid, int mode, long size)  {
389            }
390        };
391
392        try {
393            for (DataProducer data : producers) {
394                data.produce(receiver);
395            }
396        } catch(Exception e) {
397            //
398        }
399
400        return result;
401    }

413    /**
414     * Create the debian archive with from the provided control files and data producers.
415     *
416     * @param compression   the compression method used for the data file (gzip, bzip2 or anything else for no compression)
417     * @param signatureGenerator   the signature generator
418     *
419     * @return PackageDescriptor
420     * @throws PackagingException
421     */
422    public BinaryPackageControlFile createSignedDeb(Compression compression, final PGPSignatureGenerator signatureGenerator, PGPSigner signer ) throws PackagingException {
423        File tempData = null;
424        File tempControl = null;
425
426        try {
427            tempData = File.createTempFile("deb", "data");
428            tempControl = File.createTempFile("deb", "control");
429
430            console.debug("Building data");
431            DataBuilder dataBuilder = new DataBuilder(console);
432            StringBuilder md5s = new StringBuilder();
433            BigInteger size = dataBuilder.buildData(dataProducers, tempData, md5s, compression);
434
435            console.info("Building conffiles");
436            List<String> tempConffiles = populateConffiles(conffilesProducers);
437
438            console.debug("Building control");
439            ControlBuilder controlBuilder = new ControlBuilder(console, variableResolver, openReplaceToken, closeReplaceToken);
440            BinaryPackageControlFile packageControlFile = controlBuilder.createPackageControlFile(new File(control, "control"), size);
441            if (packageControlFile.get("Package") == null) {
442                packageControlFile.set("Package", packageName);
443            }
444            if (packageControlFile.get("Depends") == null) {
445                packageControlFile.set("Depends", depends);
446            }
447            if (packageControlFile.get("Section") == null) {
448                packageControlFile.set("Section", section);
449            }
450            if (packageControlFile.get("Description") == null) {
451                packageControlFile.set("Description", description);
452            }
453            if (packageControlFile.get("Homepage") == null) {
454                packageControlFile.set("Homepage", homepage);
455            }
456
457            controlBuilder.buildControl(packageControlFile, control.listFiles(), tempConffiles , md5s, tempControl);
458
459            if (!packageControlFile.isValid()) {
460                throw new PackagingException("Control file fields are invalid " + packageControlFile.invalidFields() +
461                        ". The following fields are mandatory: " + packageControlFile.getMandatoryFields() +
462                        ". Please check your pom.xml/build.xml and your control file.");
463            }
464
465            deb.getParentFile().mkdirs();
466
467            ArArchiveOutputStream ar = new ArArchiveOutputStream(new FileOutputStream(deb));
468
469            String binaryName = "debian-binary";
470            String binaryContent = "2.0\n";
471            String controlName = "control.tar.gz";
472            String dataName = "data.tar" + compression.getExtension();
473
474            addTo(ar, binaryName, binaryContent);
475            addTo(ar, controlName, tempControl);
476            addTo(ar, dataName, tempData);
477
478            if (signatureGenerator != null) {
479                console.info("Signing package with key " + key);
480
481                if(signRole == null) {
482                    signRole = "origin";
483                }
484
485                // Use debsig-verify as default
486                if(signMethod == null || !"dpkg-sig".equals(signMethod)) {
487                    // Sign file to verify with debsig-verify
488                    PGPSignatureOutputStream sigStream = new PGPSignatureOutputStream(signatureGenerator);
489
490                    addTo(sigStream, binaryContent);
491                    addTo(sigStream, tempControl);
492                    addTo(sigStream, tempData);
493                    addTo(ar, "_gpg" + signRole, sigStream.generateASCIISignature());
494
495                } else {
496
497                    // Sign file to verify with dpkg-sig --verify
498                    final String outputStr =
499                            "Version: 4\n" +
500                                    "Signer: \n" +
501                                    "Date: " + new SimpleDateFormat("EEE MMM dd HH:mm:ss yyyy", Locale.ENGLISH).format(new Date()) + "\n" +
502                                    "Role: " + signRole +"\n" +
503                                    "Files: \n" +
504                                    addFile(binaryName, binaryContent) +
505                                    addFile(controlName, tempControl) +
506                                    addFile(dataName, tempData);
507
508                    ByteArrayOutputStream message = new ByteArrayOutputStream();
509                    signer.clearSign(outputStr, message);
510
511                    addTo(ar, "_gpg" + signRole, message.toString());
512                }
513            }
514
515            ar.close();
516
517            return packageControlFile;
518
519        } catch (Exception e) {
520            throw new PackagingException("Could not create deb package", e);
521        } finally {
522            if (tempData != null) {
523                if (!tempData.delete()) {
524                    console.warn("Could not delete the temporary file " + tempData);
525                }
526            }
527            if (tempControl != null) {
528                if (!tempControl.delete()) {
529                    console.warn("Could not delete the temporary file " + tempControl);
530                }
531            }
532        }
533    }

562    private String md5Hash(byte input[]){
563        //update the input of MD5
564        MD5Digest md5 = new MD5Digest();
565        md5.update(input, 0, input.length);
566
567        //get the output/ digest size and hash it
568        byte[] digest = new byte[md5.getDigestSize()];
569        md5.doFinal(digest, 0);
570
571        return new String(Hex.encode(digest));
572    }

589    private String sha1Hash(byte input[]){
590        try
591        {
592            //prepare the input
593            MessageDigest hash = MessageDigest.getInstance("SHA1");
594            hash.update(input);
595
596            //proceed ....
597            byte[] digest = hash.digest();
598
599            return new String(Hex.encode(digest));
600        }
601        catch (NoSuchAlgorithmException e)
602        {
603            System.err.println("No such algorithm");
604            e.printStackTrace();
605        }
606
607        return null;
608    }

610    private void addTo(ArArchiveOutputStream pOutput, String pName, String pContent) throws IOException {
611        final byte[] content = pContent.getBytes();
612        pOutput.putArchiveEntry(new ArArchiveEntry(pName, content.length));
613        pOutput.write(content);
614        pOutput.closeArchiveEntry();
615    }

617    private void addTo(ArArchiveOutputStream pOutput, String pName, File pContent) throws IOException {
618        pOutput.putArchiveEntry(new ArArchiveEntry(pName, pContent.length()));
619
620        final InputStream input = new FileInputStream(pContent);
621        try {
622            Utils.copy(input, pOutput);
623        } finally {
624            input.close();
625        }
626
627        pOutput.closeArchiveEntry();
628    }
