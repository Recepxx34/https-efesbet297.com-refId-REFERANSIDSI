41  /**
42   * Utils to do the signing with OpenPGP
43   *
44   * @author Torsten Curdt
45   */
46  public final class SigningUtils {
47  
48      private static final byte[] EOL = "\r\n".getBytes(Charset.forName("UTF-8"));
49  
50      /**
51       * Create a clear sign signature over the input data. (Not detached)
52       *
53       * @param input      the content to be signed
54       * @param keyring    the keyring
55       * @param keyId      the 4 bytes identifier of the key, in hexadecimal format
56       * @param passphrase the passphrase to retrieve the private key
57       * @param output     the output destination of the signature
58       */
59      public static void clearSign(InputStream input, InputStream keyring, String keyId, String passphrase, OutputStream output) throws IOException, PGPException, GeneralSecurityException {
60          PGPSecretKey secretKey = getSecretKey(keyring, keyId);
61          PGPPrivateKey privateKey = secretKey.extractPrivateKey(new BcPBESecretKeyDecryptorBuilder(new BcPGPDigestCalculatorProvider()).build(passphrase.toCharArray()));
62  
63          int digest = PGPUtil.SHA1;
64  
65          PGPSignatureGenerator signatureGenerator = new PGPSignatureGenerator(new BcPGPContentSignerBuilder(secretKey.getPublicKey().getAlgorithm(), digest));
66          signatureGenerator.init(PGPSignature.CANONICAL_TEXT_DOCUMENT, privateKey);
67  
68          ArmoredOutputStream armoredOutput = new ArmoredOutputStream(output);
69          armoredOutput.beginClearText(digest);
70  
71          BufferedReader reader = new BufferedReader(new InputStreamReader(input));
72  
73          String line;
74          while ((line = reader.readLine()) != null) {
75              // trailing spaces must be removed for signature calculation (see http://tools.ietf.org/html/rfc4880#section-7.1)
76              byte[] data = trim(line).getBytes("UTF-8");
77  
78              armoredOutput.write(data);
79              armoredOutput.write(EOL);
80  
81              signatureGenerator.update(data);
82              signatureGenerator.update(EOL);
83          }
84  
85          armoredOutput.endClearText();
86  
87          PGPSignature signature = signatureGenerator.generate();
88          signature.encode(new BCPGOutputStream(armoredOutput));
89  
90          armoredOutput.close();
91      }
92  
93      /**
94       * Returns the secret key matching the specified identifier.
95       * 
96       * @param input the input stream containing the keyring collection
97       * @param keyId the 4 bytes identifier of the key
98       */
99      private static PGPSecretKey getSecretKey(InputStream input, String keyId) throws IOException, PGPException {
100         PGPSecretKeyRingCollection keyrings = new PGPSecretKeyRingCollection(PGPUtil.getDecoderStream(input));
101 
102         Iterator rIt = keyrings.getKeyRings();
103 
104         while (rIt.hasNext()) {
105             PGPSecretKeyRing kRing = (PGPSecretKeyRing) rIt.next();
106             Iterator kIt = kRing.getSecretKeys();
107 
108             while (kIt.hasNext()) {
109                 PGPSecretKey key = (PGPSecretKey) kIt.next();
110 
111                 if (key.isSigningKey() && Long.toHexString(key.getKeyID() & 0xFFFFFFFFL).equals(keyId.toLowerCase())) {
112                     return key;
113                 }
114             }
115         }
116 
117         return null;
118     }
119 
120     /**
121      * Trim the trailing spaces.
122      * 
123      * @param line
124      */
125     private static String trim(String line) {
126         char[] chars = line.toCharArray();
127         int len = chars.length;
128 
129         while (len > 0) {
130             if (!Character.isWhitespace(chars[len - 1])) {
131                 break;
132             }
133             len--;
134         }
135 
136         return line.substring(0, len);
137     }
