220     public void validate() throws PackagingException {
221         if (control == null || !control.isDirectory()) {
222             throw new PackagingException("The 'control' attribute doesn't point to a directory.");
223         }
224 
225         if (changesIn != null) {
226 
227             if (changesIn.exists() && (!changesIn.isFile() || !changesIn.canRead())) {
228                 throw new PackagingException("The 'changesIn' setting needs to point to a readable file. " + changesIn + " was not found/readable.");
229             }
230 
231             if (changesOut != null && !isWritableFile(changesOut)) {
232                 throw new PackagingException("Cannot write the output for 'changesOut' to " + changesOut);
233             }
234 
235             if (changesSave != null && !isWritableFile(changesSave)) {
236                 throw new PackagingException("Cannot write the output for 'changesSave' to " + changesSave);
237             }
238 
239         } else {
240             if (changesOut != null || changesSave != null) {
241                 throw new PackagingException("The 'changesOut' or 'changesSave' settings may only be used when there is a 'changesIn' specified.");
242             }
243         }
244 
245         if (Compression.toEnum(compression) == null) {
246             throw new PackagingException("The compression method '" + compression + "' is not supported (expected 'none', 'gzip', 'bzip2' or 'xz')");
247         }
248 
249         if (deb == null) {
250             throw new PackagingException("You need to specify where the deb file is supposed to be created.");
251         }
252     }

254     public void makeDeb() throws PackagingException {
255         BinaryPackageControlFile packageControlFile;
256         try {
257             console.info("Creating debian package: " + deb);
258 
259             // If we should sign the package
260             boolean doSign = signPackage;
261 
262             if (doSign) {
263 
264                 if (keyring == null || !keyring.exists()) {
265                     doSign = false;
266                     console.warn("Signing requested, but no keyring supplied");
267                 }
268 
269                 if (key == null) {
270                     doSign = false;
271                     console.warn("Signing requested, but no key supplied");
272                 }
273 
274                 if (passphrase == null) {
275                     doSign = false;
276                     console.warn("Signing requested, but no passphrase supplied");
277                 }
278 
279                 FileInputStream keyRingInput = new FileInputStream(keyring);
280                 PGPSigner signer = null;
281                 try {
282                     signer = new PGPSigner(new FileInputStream(keyring), key, passphrase);
283                 } finally {
284                     keyRingInput.close();
285                 }
286 
287                 int digest = PGPUtil.SHA1;
288 
289                 PGPSignatureGenerator signatureGenerator = new PGPSignatureGenerator(new BcPGPContentSignerBuilder(signer.getSecretKey().getPublicKey().getAlgorithm(), digest));
290                 signatureGenerator.init(PGPSignature.BINARY_DOCUMENT, signer.getPrivateKey());
291 
292                 packageControlFile = createSignedDeb(Compression.toEnum(compression), signatureGenerator, signer);
293             } else {
294                 packageControlFile = createDeb(Compression.toEnum(compression));
295             }
296 
297         } catch (Exception e) {
298             throw new PackagingException("Failed to create debian package " + deb, e);
299         }
300 
301         makeChangesFiles(packageControlFile);
302     }

304     private void makeChangesFiles(final BinaryPackageControlFile packageControlFile) throws PackagingException {
305         if (changesOut == null) {
306             changesOut = new File(deb.getParentFile(), deb.getName().replace(".deb", ".changes"));
307         }
308 
309         ChangesProvider changesProvider;
310         FileOutputStream out = null;
311 
312         try {
313             console.info("Creating changes file: " + changesOut);
314 
315             out = new FileOutputStream(changesOut);
316 
317             if (changesIn != null && changesIn.exists()) {
318                 // read the changes form a textfile provider
319                 changesProvider = new TextfileChangesProvider(new FileInputStream(changesIn), packageControlFile);
320             } else {
321                 // create an empty changelog
322                 changesProvider = new ChangesProvider() {
323                     @Override
324                     public ChangeSet[] getChangesSets() {
325                         return new ChangeSet[] {
326                                 new ChangeSet(packageControlFile.get("Package"),
327                                         packageControlFile.get("Version"),
328                                         new Date(),
329                                         packageControlFile.get("Distribution"),
330                                         packageControlFile.get("Urgency"),
331                                         packageControlFile.get("Maintainer"),
332                                         new String[0])
333                         };
334                     }
335                 };
336             }
337 
338             ChangesFileBuilder builder = new ChangesFileBuilder();
339             ChangesFile changesFile = builder.createChanges(packageControlFile, deb, changesProvider);
340 
341             if (keyring != null && key != null && passphrase != null) {
342                 console.info("Signing the changes file with the key " + key);
343                 PGPSigner signer = new PGPSigner(new FileInputStream(keyring), key, passphrase);
344                 signer.clearSign(changesFile.toString(), out);
345             } else {
346                 out.write(changesFile.toString().getBytes("UTF-8"));
347             }
348             out.flush();
349 
350         } catch (Exception e) {
351             throw new PackagingException("Failed to create the Debian changes file " + changesOut, e);
352         } finally {
353             IOUtils.closeQuietly(out);
354         }
355 
356         if (changesSave == null || !(changesProvider instanceof TextfileChangesProvider)) {
357             return;
358         }
359 
360         try {
361             console.info("Saving changes to file: " + changesSave);
362 
363             ((TextfileChangesProvider) changesProvider).save(new FileOutputStream(changesSave));
364 
365         } catch (Exception e) {
366             throw new PackagingException("Failed to save debian changes file " + changesSave, e);
367         }
368     }

370     private List<String> populateConffiles(Collection<DataProducer> producers) {
371         final List<String> result = new ArrayList<String>();
372 
373         if (producers == null || producers.isEmpty()) {
374             return result;
375         }
376 
377         final DataConsumer receiver = new DataConsumer() {
378             public void onEachDir( String dirname, String linkname, String user, int uid, String group, int gid, int mode, long size ) throws IOException {
379                 //
380             }
381 
382             public void onEachFile( InputStream inputStream, String filename, String linkname, String user, int uid, String group, int gid, int mode, long size ) throws IOException {
383                 String tempConffileItem = filename;
384                 if (tempConffileItem.startsWith(".")) {
385                     tempConffileItem = tempConffileItem.substring(1);
386                 }
387                 console.info("Adding conffile: " + tempConffileItem);
388                 result.add(tempConffileItem);
389             }
390 
391             public void onEachLink(String path, String linkname, boolean symlink, String user, int uid, String group, int gid, int mode) throws IOException {
392                 //
393             }
394         };
395 
396         try {
397             for (DataProducer data : producers) {
398                 data.produce(receiver);
399             }
400         } catch(Exception e) {
401             //
402         }
403 
404         return result;
405     }

420     /**
421      * Create the debian archive with from the provided control files and data producers.
422      *
423      * @param compression   the compression method used for the data file (gzip, bzip2 or anything else for no compression)
424      * @param signatureGenerator   the signature generator
425      *
426      * @return PackageDescriptor
427      * @throws PackagingException
428      */
429     public BinaryPackageControlFile createSignedDeb(Compression compression, final PGPSignatureGenerator signatureGenerator, PGPSigner signer ) throws PackagingException {
430         File tempData = null;
431         File tempControl = null;
432 
433         try {
434             tempData = File.createTempFile("deb", "data");
435             tempControl = File.createTempFile("deb", "control");
436 
437             console.debug("Building data");
438             DataBuilder dataBuilder = new DataBuilder(console);
439             StringBuilder md5s = new StringBuilder();
440             BigInteger size = dataBuilder.buildData(dataProducers, tempData, md5s, compression);
441 
442             console.info("Building conffiles");
443             List<String> tempConffiles = populateConffiles(conffilesProducers);
444 
445             console.debug("Building control");
446             ControlBuilder controlBuilder = new ControlBuilder(console, variableResolver, openReplaceToken, closeReplaceToken);
447             BinaryPackageControlFile packageControlFile = controlBuilder.createPackageControlFile(new File(control, "control"), size);
448             if (packageControlFile.get("Package") == null) {
449                 packageControlFile.set("Package", packageName);
450             }
451             if (packageControlFile.get("Depends") == null) {
452                 packageControlFile.set("Depends", depends);
453             }
454             if (packageControlFile.get("Section") == null) {
455                 packageControlFile.set("Section", section);
456             }
457             if (packageControlFile.get("Description") == null) {
458                 packageControlFile.set("Description", description);
459             }
460             if (packageControlFile.get("Homepage") == null) {
461                 packageControlFile.set("Homepage", homepage);
462             }
463 
464             controlBuilder.buildControl(packageControlFile, control.listFiles(), tempConffiles , md5s, tempControl);
465 
466             if (!packageControlFile.isValid()) {
467                 throw new PackagingException("Control file fields are invalid " + packageControlFile.invalidFields() +
468                         ". The following fields are mandatory: " + packageControlFile.getMandatoryFields() +
469                         ". Please check your pom.xml/build.xml and your control file.");
470             }
471 
472             deb.getParentFile().mkdirs();
473 
474             ArArchiveOutputStream ar = new ArArchiveOutputStream(new FileOutputStream(deb));
475 
476             String binaryName = "debian-binary";
477             String binaryContent = "2.0\n";
478             String controlName = "control.tar.gz";
479             String dataName = "data.tar" + compression.getExtension();
480 
481             addTo(ar, binaryName, binaryContent);
482             addTo(ar, controlName, tempControl);
483             addTo(ar, dataName, tempData);
484 
485             if (signatureGenerator != null) {
486                 console.info("Signing package with key " + key);
487 
488                 if(signRole == null) {
489                     signRole = "origin";
490                 }
491 
492                 // Use debsig-verify as default
493                 if(signMethod == null || !"dpkg-sig".equals(signMethod)) {
494                     // Sign file to verify with debsig-verify
495                     PGPSignatureOutputStream sigStream = new PGPSignatureOutputStream(signatureGenerator);
496 
497                     addTo(sigStream, binaryContent);
498                     addTo(sigStream, tempControl);
499                     addTo(sigStream, tempData);
500                     addTo(ar, "_gpg" + signRole, sigStream.generateASCIISignature());
501 
502                 } else {
503 
504                     // Sign file to verify with dpkg-sig --verify
505                     final String outputStr =
506                             "Version: 4\n" +
507                                     "Signer: \n" +
508                                     "Date: " + new SimpleDateFormat("EEE MMM dd HH:mm:ss yyyy", Locale.ENGLISH).format(new Date()) + "\n" +
509                                     "Role: " + signRole +"\n" +
510                                     "Files: \n" +
511                                     addFile(binaryName, binaryContent) +
512                                     addFile(controlName, tempControl) +
513                                     addFile(dataName, tempData);
514 
515                     ByteArrayOutputStream message = new ByteArrayOutputStream();
516                     signer.clearSign(outputStr, message);
517 
518                     addTo(ar, "_gpg" + signRole, message.toString());
519                 }
520             }
521 
522             ar.close();
523 
524             return packageControlFile;
525 
526         } catch (Exception e) {
527             throw new PackagingException("Could not create deb package", e);
528         } finally {
529             if (tempData != null) {
530                 if (!tempData.delete()) {
531                     console.warn("Could not delete the temporary file " + tempData);
532                 }
533             }
534             if (tempControl != null) {
535                 if (!tempControl.delete()) {
536                     console.warn("Could not delete the temporary file " + tempControl);
537                 }
538             }
539         }
540     }

569     private String md5Hash(byte input[]){
570         //update the input of MD5
571         MD5Digest md5 = new MD5Digest();
572         md5.update(input, 0, input.length);
573 
574         //get the output/ digest size and hash it
575         byte[] digest = new byte[md5.getDigestSize()];
576         md5.doFinal(digest, 0);
577 
578         return new String(Hex.encode(digest));
579     }

596     private String sha1Hash(byte input[]){
597         try
598         {
599             //prepare the input
600             MessageDigest hash = MessageDigest.getInstance("SHA1");
601             hash.update(input);
602 
603             //proceed ....
604             byte[] digest = hash.digest();
605 
606             return new String(Hex.encode(digest));
607         }
608         catch (NoSuchAlgorithmException e)
609         {
610             System.err.println("No such algorithm");
611             e.printStackTrace();
612         }
