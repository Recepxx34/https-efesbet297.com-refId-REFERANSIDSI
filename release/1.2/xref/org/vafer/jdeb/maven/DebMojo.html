342     protected VariableResolver initializeVariableResolver( Map<String, String> variables ) {
343         @SuppressWarnings("unchecked")
344         final Map<String, String> projectProperties = Map.class.cast(getProject().getProperties());
345         @SuppressWarnings("unchecked")
346         final Map<String, String> systemProperties = Map.class.cast(System.getProperties());
347 
348         variables.putAll(projectProperties);
349         variables.putAll(systemProperties);
350         variables.put("name", name != null ? name : getProject().getName());
351         variables.put("artifactId", getProject().getArtifactId());
352         variables.put("groupId", getProject().getGroupId());
353         variables.put("version", getProjectVersion());
354         variables.put("description", getProject().getDescription());
355         variables.put("extension", "deb");
356         variables.put("baseDir", getProject().getBasedir().getAbsolutePath());
357         variables.put("buildDir", buildDirectory.getAbsolutePath());
358         variables.put("project.version", getProject().getVersion());
359         variables.put("url", getProject().getUrl());
360 
361         return new MapVariableResolver(variables);
362     }

409     /**
410      * Main entry point
411      *
412      * @throws MojoExecutionException on error
413      */
414     public void execute() throws MojoExecutionException {
415 
416         final MavenProject project = getProject();
417 
418         if (skip) {
419             getLog().info("skipping as configured (skip)");
420             return;
421         }
422 
423         if (skipPOMs && isPOM()) {
424             getLog().info("skipping because artifact is a pom (skipPOMs)");
425             return;
426         }
427 
428         if (skipSubmodules && isSubmodule()) {
429             getLog().info("skipping submodule (skipSubmodules)");
430             return;
431         }
432 
433 
434         setData(dataSet);
435 
436         console = new MojoConsole(getLog(), verbose);
437 
438         initializeSignProperties();
439 
440         final VariableResolver resolver = initializeVariableResolver(new HashMap<String, String>());
441 
442         final File debFile = new File(Utils.replaceVariables(resolver, deb, openReplaceToken, closeReplaceToken));
443         final File controlDirFile = new File(Utils.replaceVariables(resolver, controlDir, openReplaceToken, closeReplaceToken));
444         final File installDirFile = new File(Utils.replaceVariables(resolver, installDir, openReplaceToken, closeReplaceToken));
445         final File changesInFile = new File(Utils.replaceVariables(resolver, changesIn, openReplaceToken, closeReplaceToken));
446         final File changesOutFile = new File(Utils.replaceVariables(resolver, changesOut, openReplaceToken, closeReplaceToken));
447         final File changesSaveFile = new File(Utils.replaceVariables(resolver, changesSave, openReplaceToken, closeReplaceToken));
448         final File keyringFile = keyring == null ? null : new File(Utils.replaceVariables(resolver, keyring, openReplaceToken, closeReplaceToken));
449 
450         // if there are no producers defined we try to use the artifacts
451         if (dataProducers.isEmpty()) {
452 
453             if (hasMainArtifact()) {
454                 Set<Artifact> artifacts = new HashSet<Artifact>();
455 
456                 artifacts.add(project.getArtifact());
457 
458                 @SuppressWarnings("unchecked")
459                 final Set<Artifact> projectArtifacts = project.getArtifacts();
460 
461                 for (Artifact artifact : projectArtifacts) {
462                     artifacts.add(artifact);
463                 }
464 
465                 @SuppressWarnings("unchecked")
466                 final List<Artifact> attachedArtifacts = project.getAttachedArtifacts();
467 
468                 for (Artifact artifact : attachedArtifacts) {
469                     artifacts.add(artifact);
470                 }
471 
472                 for (Artifact artifact : artifacts) {
473                     final File file = artifact.getFile();
474                     if (file != null) {
475                         dataProducers.add(new DataProducer() {
476                             @Override
477                             public void produce( final DataConsumer receiver ) {
478                                 try {
479                                     receiver.onEachFile(
480                                         new FileInputStream(file),
481                                         new File(installDirFile, file.getName()).getAbsolutePath(),
482                                         "",
483                                         "root", 0, "root", 0,
484                                         TarEntry.DEFAULT_FILE_MODE,
485                                         file.length());
486                                 } catch (Exception e) {
487                                     getLog().error(e);
488                                 }
489                             }
490                         });
491                     } else {
492                         getLog().error("No file for artifact " + artifact);
493                     }
494                 }
495             }
496         }
497 
498         try {
499             DebMaker debMaker = new DebMaker(console, dataProducers, conffileProducers);
500             debMaker.setDeb(debFile);
501             debMaker.setControl(controlDirFile);
502             debMaker.setPackage(getProject().getArtifactId());
503             debMaker.setDescription(getProject().getDescription());
504             debMaker.setHomepage(getProject().getUrl());
505             debMaker.setChangesIn(changesInFile);
506             debMaker.setChangesOut(changesOutFile);
507             debMaker.setChangesSave(changesSaveFile);
508             debMaker.setCompression(compression);
509             debMaker.setKeyring(keyringFile);
510             debMaker.setKey(key);
511             debMaker.setPassphrase(passphrase);
512             debMaker.setSignPackage(signPackage);
513             debMaker.setSignMethod(signMethod);
514             debMaker.setSignRole(signRole);
515             debMaker.setResolver(resolver);
516             debMaker.setOpenReplaceToken(openReplaceToken);
517             debMaker.setCloseReplaceToken(closeReplaceToken);
518             debMaker.validate();
519             debMaker.makeDeb();
520 
521             // Always attach unless explicitly set to false
522             if ("true".equalsIgnoreCase(attach)) {
523                 console.info("Attaching created debian package " + debFile);
524                 if (!isType()) {
525                     projectHelper.attachArtifact(project, type, classifier, debFile);
526                 } else {
527                     project.getArtifact().setFile(debFile);
528                 }
529             }
530 
531         } catch (PackagingException e) {
532             getLog().error("Failed to create debian package " + debFile, e);
533             throw new MojoExecutionException("Failed to create debian package " + debFile, e);
534         }
535     }

537     /**
538      * Initializes unspecified sign properties using available defaults
539      * and global settings.
540      */
541     private void initializeSignProperties() {
542         if (!signPackage) {
543             return;
544         }
545 
546         if (key != null && keyring != null && passphrase != null) {
547             return;
548         }
549 
550         Map<String, String> properties =
551                 readPropertiesFromActiveProfiles(signCfgPrefix, KEY, KEYRING, PASSPHRASE);
552 
553         key = lookupIfEmpty(key, properties, KEY);
554         keyring = lookupIfEmpty(keyring, properties, KEYRING);
555         passphrase = decrypt(lookupIfEmpty(passphrase, properties, PASSPHRASE));
556 
557         if (keyring == null) {
558             try {
559                 keyring = Utils.guessKeyRingFile().getAbsolutePath();
560                 console.info("Located keyring at " + keyring);
561             } catch (FileNotFoundException e) {
562                 console.warn(e.getMessage());
563             }
564         }
565     }

567     /**
568      * Decrypts given passphrase if needed using maven security dispatcher.
569      * See http://maven.apache.org/guides/mini/guide-encryption.html for details.
570      *
571      * @param maybeEncryptedPassphrase possibly encrypted passphrase
572      * @return decrypted passphrase
573      */
574     private String decrypt( final String maybeEncryptedPassphrase ) {
575         if (maybeEncryptedPassphrase == null) {
576             return null;
577         }
578 
579         try {
580             final String decrypted = secDispatcher.decrypt(maybeEncryptedPassphrase);
581             if (maybeEncryptedPassphrase.equals(decrypted)) {
582                 console.info("Passphrase was not encrypted");
583             } else {
584                 console.info("Passphrase was successfully decrypted");
585             }
586             return decrypted;
587         } catch (SecDispatcherException e) {
588             console.warn("Unable to decrypt passphrase: " + e.getMessage());
589         }
590 
591         return maybeEncryptedPassphrase;
592     }

595     /**
596      * Read properties from the active profiles.
597      *
598      * Goes through all active profiles (in the order the
599      * profiles are defined in settings.xml) and extracts
600      * the desired properties (if present). The prefix is
601      * used when looking up properties in the profile but
602      * not in the returned map.
603      *
604      * @param prefix The prefix to use or null if no prefix should be used
605      * @param properties The properties to read
606      *
607      * @return A map containing the values for the properties that were found
608      */
609     public Map<String, String> readPropertiesFromActiveProfiles( final String prefix,
610                                                                  final String... properties ) {
611         if (settings == null) {
612             console.debug("No maven setting injected");
613             return Collections.emptyMap();
614         }
615 
616         final List<String> activeProfilesList = settings.getActiveProfiles();
617         if (activeProfilesList.isEmpty()) {
618             console.debug("No active profiles found");
619             return Collections.emptyMap();
620         }
621 
622         final Map<String, String> map = new HashMap<String, String>();
623         final Set<String> activeProfiles = new HashSet<String>(activeProfilesList);
624 
625         // Iterate over all active profiles in order
626         for (final Profile profile : settings.getProfiles()) {
627             // Check if the profile is active
628             final String profileId = profile.getId();
629             if (activeProfiles.contains(profileId)) {
630                 console.debug("Trying active profile " + profileId);
631                 for (final String property : properties) {
632                     final String propKey = prefix != null ? prefix + property : property;
633                     final String value = profile.getProperties().getProperty(propKey);
634                     if (value != null) {
635                         console.debug("Found property " + property + " in profile " + profileId);
636                         map.put(property, value);
637                     }
638                 }
639             }
640         }
641 
642         return map;
643     }
