016    package org.vafer.jdeb.ant;
017    
018    import java.io.File;
019    import java.io.FileInputStream;
020    import java.io.FileOutputStream;
021    import java.util.ArrayList;
022    import java.util.Collection;
023    
024    import org.apache.tools.ant.BuildException;
025    import org.apache.tools.ant.taskdefs.MatchingTask;
026    import org.apache.tools.ant.taskdefs.Tar;
027    import org.apache.tools.ant.types.FileSet;
028    import org.vafer.jdeb.DataProducer;
029    import org.vafer.jdeb.Processor;
030    import org.vafer.jdeb.changes.TextfileChangesProvider;
031    import org.vafer.jdeb.descriptors.PackageDescriptor;
032    import org.vafer.jdeb.producers.DataProducerFileSet;
033    
034    /**
035     * TODO generalize with DebMaker
036     *
037     * AntTask for creating debian archives.
038     * Even supports signed changes files.
039     *
040     * @author Torsten Curdt
041     */
042    
043    public class DebAntTask extends MatchingTask {
044    
045        /**
046         * The Debian package produced
047         */
048        private File deb;
049    
050        /**
051         * The directory containing the control files to build the package
052         */
053        private File control;
054    
055        /**
056         * The file containing the PGP keys
057         */
058        private File keyring;
059    
060        /**
061         * The key to use in the keyring
062         */
063        private String key;
064    
065        /**
066         * The passphrase for the key to sign the changes file
067         */
068        private String passphrase;
069    
070        /**
071         * The file to read the changes from
072         */
073        private File changesIn;
074    
075        /**
076         * The file where to write the changes to
077         */
078        private File changesOut;
079    
080        /**
081         * The file where to write the changes of the changes input to
082         */
083        private File changesSave;
084    
085        /**
086         * The compression method used for the data file (none, gzip or bzip2)
087         */
088        private String compression = "gzip";
089    
090        /**
091         * Trigger the verbose mode detailing all operations
092         */
093        private boolean verbose;
094    
095        private Collection dataProducers = new ArrayList();
096    
097    
098        public void setDestfile( File deb ) {
099            this.deb = deb;
100        }
101    
102        public void setControl( File control ) {
103            this.control = control;
104        }
105    
106        public void setChangesIn( File changes ) {
107            this.changesIn = changes;
108        }
109    
110        public void setChangesOut( File changes ) {
111            this.changesOut = changes;
112        }
113    
114        public void setChangesSave( File changes ) {
115            this.changesSave = changes;
116        }
117    
118        public void setKeyring( File keyring ) {
119            this.keyring = keyring;
120        }
121    
122        public void setKey( String key ) {
123            this.key = key;
124        }
125    
126        public void setPassphrase( String passphrase ) {
127            this.passphrase = passphrase;
128        }
129    
130        public void setCompression( String compression ) {
131            this.compression = compression;
132        }
133    
134        public void setVerbose( boolean verbose ) {
135            this.verbose = verbose;
136        }
137    
138        public void addFileSet( FileSet fileset ) {
139            dataProducers.add(new DataProducerFileSet(fileset));
140        }
141    
142        public void addTarFileSet( Tar.TarFileSet fileset ) {
143            dataProducers.add(new DataProducerFileSet(fileset));
144        }
145    
146        public void addData( Data data ) {
147            dataProducers.add(data);
148        }
149    
150        private boolean isPossibleOutput( File file ) {
151    
152            if (file.exists()) {
153                return file.isFile() && file.canWrite();
154            }
155    
156            return true;
157        }
158    
159        public void execute() {
160    
161            if (control == null || !control.isDirectory()) {
162                throw new BuildException("You need to point the 'control' attribute to the control directory.");
163            }
164    
165            if (changesIn != null) {
166    
167                if (!changesIn.isFile() || !changesIn.canRead()) {
168                    throw new BuildException("The 'changesIn' attribute needs to point to a readable file. " + changesIn + " was not found/readable.");
169                }
170    
171                if (changesOut == null) {
172                    throw new BuildException("A 'changesIn' without a 'changesOut' does not make much sense.");
173                }
174    
175                if (!isPossibleOutput(changesOut)) {
176                    throw new BuildException("Cannot write the output for 'changesOut' to " + changesOut);
177                }
178    
179                if (changesSave != null && !isPossibleOutput(changesSave)) {
180                    throw new BuildException("Cannot write the output for 'changesSave' to " + changesSave);
181                }
182    
183            } else {
184                if (changesOut != null || changesSave != null) {
185                    throw new BuildException("The 'changesOut' or 'changesSave' attributes may only be used when there is a 'changesIn' specified.");
186                }
187            }
188    
189            if (!"gzip".equals(compression) && !"bzip2".equals(compression) && !"none".equals(compression)) {
190                throw new BuildException("The compression method '" + compression + "' is not supported");
191            }
192    
193            if (dataProducers.size() == 0) {
194                throw new BuildException("You need to provide at least one reference to a tgz or directory with data.");
195            }
196    
197            if (deb == null) {
198                throw new BuildException("You need to point the 'destfile' attribute to where the deb is supposed to be created.");
199            }
200    
201            final File[] controlFiles = control.listFiles();
202    
203            final DataProducer[] data = new DataProducer[dataProducers.size()];
204            dataProducers.toArray(data);
205    
206            final Processor processor = new Processor(new TaskConsole(this, verbose), null);
207    
208            final PackageDescriptor packageDescriptor;
209            try {
210    
211                log("Creating debian package: " + deb);
212    
213                packageDescriptor = processor.createDeb(controlFiles, data, deb, compression);
214    
215            } catch (Exception e) {
216                // what the fuck ant? why are you not printing the exception chain?
217                e.printStackTrace();
218                throw new BuildException("Failed to create debian package " + deb, e);
219            }
220    
221            final TextfileChangesProvider changesProvider;
222    
223            try {
224                if (changesOut == null) {
225                    return;
226                }
227    
228                log("Creating changes file: " + changesOut);
229    
230                // for now only support reading the changes form a textfile provider
231                changesProvider = new TextfileChangesProvider(new FileInputStream(changesIn), packageDescriptor);
232    
233                processor.createChanges(packageDescriptor, changesProvider, (keyring != null) ? new FileInputStream(keyring) : null, key, passphrase, new FileOutputStream(changesOut));
234    
235            } catch (Exception e) {
236                throw new BuildException("Failed to create debian changes file " + changesOut, e);
237            }
238    
239            try {
240                if (changesSave == null) {
241                    return;
242                }
243    
244                log("Saving changes to file: " + changesSave);
245    
246                changesProvider.save(new FileOutputStream(changesSave));
247    
248            } catch (Exception e) {
249                throw new BuildException("Failed to save debian changes file " + changesSave, e);
250            }
251    
252        }
253    }
