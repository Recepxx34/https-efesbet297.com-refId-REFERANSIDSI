001package org.vafer.jdeb.utils;
002
003import java.io.ByteArrayOutputStream;
004import java.io.IOException;
005import java.io.OutputStream;
006import java.security.SignatureException;
007
008import org.bouncycastle.bcpg.ArmoredOutputStream;
009import org.bouncycastle.openpgp.PGPException;
010import org.bouncycastle.openpgp.PGPSignature;
011import org.bouncycastle.openpgp.PGPSignatureGenerator;
012
013/**
014 * An output stream that calculates the signature of the input data as it
015 * is written
016 * 
017 * @author mpoindexter
018 *
019 */
020public class PGPSignatureOutputStream extends OutputStream {
021    private final PGPSignatureGenerator signatureGenerator;
022
023    public PGPSignatureOutputStream( PGPSignatureGenerator signatureGenerator ) {
024        super();
025        this.signatureGenerator = signatureGenerator;
026    }
027
028    public void write( int b ) throws IOException {
029        try {
030            signatureGenerator.update(new byte[] { (byte)b });
031        } catch(SignatureException e) {
032            throw new IOException(e);
033        }
034    }
035
036    public void write( byte[] b ) throws IOException {
037        try {
038            signatureGenerator.update(b);
039        } catch(SignatureException e) {
040            throw new IOException(e);
041        }
042    }
043
044    public void write( byte[] b, int off, int len ) throws IOException {
045        try {
046            signatureGenerator.update(b, off, len);
047        } catch(SignatureException e) {
048            throw new IOException(e);
049        }
050    }
051    
052    public PGPSignature generateSignature() throws SignatureException, PGPException {
053        return signatureGenerator.generate();
054    }
055    
056    public String generateASCIISignature() throws SignatureException, PGPException {
057        try {
058            PGPSignature signature = generateSignature();
059            ByteArrayOutputStream buffer = new ByteArrayOutputStream();
060            ArmoredOutputStream armorStream = new ArmoredOutputStream(buffer);
061            signature.encode(armorStream);
062            armorStream.close();
063            return new String(buffer.toByteArray());
064        } catch(IOException e) {
065            //Should never happen since we are just using a memory buffer
066            throw new RuntimeException(e);
067        }
068    }
069
070}
