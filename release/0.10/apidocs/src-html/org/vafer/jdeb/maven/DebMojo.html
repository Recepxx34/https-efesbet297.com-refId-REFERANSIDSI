044    /**

052         * @component

056        /**
057         * Defines the pattern of the name of final artifacts. Possible
058         * substitutions are [[baseDir]] [[buildDir]] [[artifactId]] [[version]]
059         * [[extension]] and [[groupId]].

063        private String deb;

065        /**
066         * Explicitly defines the path to the control directory. At least the
067         * control file is mandatory.

074         * Explicitly define the file to read the changes from.
075         *
076         * @parameter default-value="[[baseDir]]/CHANGES.txt"

078        private String changesIn;
079    
080        /**
081         * Explicitly define the file where to write the changes to.
082         *
083         * @parameter default-value="[[buildDir]]/[[artifactId]]_[[version]].changes"
084         */
085        private String changesOut;
086    
087        /**
088         * Explicitly define the file where to write the changes of the changes
089         * input to.
090         *
091         * @parameter default-value="[[baseDir]]/CHANGES.txt"
092         */
093        private String changesSave;
094    
095        /**
096         * The compression method used for the data file (none, gzip or bzip2)

098         * @parameter default-value="gzip"

100        private String compression;

102

104         * Boolean option whether to attach the artifact to the project
105         *
106         *  @parameter default-value="true"

108        private String attach;

111         * The location where all package files will be installed. By default, all

113         * http://www.pathname.com/
114         * fhs/pub/fhs-2.3.html#OPTADDONAPPLICATIONSOFTWAREPACKAGES)
115         *
116         * @parameter default-value="/opt/[[artifactId]]"
117         */
118        private String installDir;

120    
121        /**
122         * The type of attached artifact
123         *
124         * @parameter default-value="deb"

126        private String type;

129         * The project base directory
130         *
131         * @parameter default-value="${basedir}"
132         * @required

135        private File baseDir;

137        /**
138         * Run the plugin on all sub-modules.
139         * If set to false, the plugin will be run in the same folder where the
140         * mvn command was invoked
141         * @parameter expression="${submodules}" default-value="true"
142         */
143        private boolean submodules;
144    
145        /**
146         * The Maven Session Object
147         *
148         * @parameter expression="${session}"
149         * @required
150         * @readonly
151         */
152        private MavenSession session;

154        /**
155         * The classifier of attached artifact
156         *
157         * @parameter
158         */
159        private String classifier;

161        /**
162         * "data" entries used to determine which files should be added to this deb.
163         * The "data" entries may specify a tarball (tar.gz, tar.bz2, tgz), a
164         * directory, or a normal file. An entry would look something like this in
165         * your pom.xml:
166         *
167         * <pre>
168         *   <build>
169         *     <plugins>
170         *       <plugin>
171         *       <artifactId>jdeb</artifactId>
172         *       <groupId>org.vafer</groupId>
173         *       ...
174         *       <configuration>
175         *         ...
176         *         <dataSet>
177         *           <data>
178         *             <src>${project.basedir}/target/my_archive.tar.gz</src>
179         *             <include>...</include>
180         *             <exclude>...</exclude>
181         *             <mapper>
182         *               <type>perm</type>
183         *               <strip>1</strip>
184         *               <prefix>/somewhere/else</prefix>
185         *               <user>santbj</user>
186         *               <group>santbj</group>
187         *               <mode>600</mode>
188         *             </mapper>
189         *           </data>
190         *           <data>
191         *             <src>${project.build.directory}/data</src>
192         *             <include></include>
193         *             <exclude>**&#47;.svn</exclude>
194         *             <mapper>
195         *               <type>ls</type>
196         *               <src>mapping.txt</src>
197         *             </mapper>
198         *           </data>
199         *         <data>
200         *           <src>${project.basedir}/README.txt</src>
201         *         </data>
202         *         </dataSet>
203         *       </configuration>
204         *     </plugins>
205         *   </build>
206         * </pre>
207         *
208         * @parameter expression="${dataSet}"
209         */
210        private Data[] dataSet;
211    
212        /**
213         * When SNAPSHOT version replace <code>SNAPSHOT</code> with current date
214         * and time to make sure each build is unique.
215         *
216         * @parameter expression="${timestamped}" default-value="false"
217         */
218        private boolean timestamped;
219    
220        /* end of parameters */
221    
222        private String openReplaceToken = "[[";
223        private String closeReplaceToken = "]]";
224        private Collection dataProducers = new ArrayList();
225    
226        public void setOpenReplaceToken(String openReplaceToken) {
227            this.openReplaceToken = openReplaceToken;
228            AbstractDescriptor.setOpenToken(openReplaceToken);
229        }
230    
231        public void setCloseReplaceToken(String closeReplaceToken) {
232            this.closeReplaceToken = closeReplaceToken;
233            AbstractDescriptor.setCloseToken(closeReplaceToken);
234        }
235    
236        protected void setData(Data[] pData) {
237            dataSet = pData;
238            dataProducers.clear();
239            if (pData != null) {
240                for (int i = 0; i < pData.length; i++) {
241                    dataProducers.add(pData[i]);
242                }
243            }
244        }
245    
246        protected VariableResolver initializeVariableResolver(Map variables) {
247            variables.putAll(getProject().getProperties());
248    
249            variables.put("name", getProject().getName());
250            variables.put("artifactId", getProject().getArtifactId());
251            variables.put("groupId", getProject().getGroupId());
252            variables.put("version", getProjectVersion());
253            variables.put("description", getProject().getDescription());
254            variables.put("extension", "deb");
255            variables.put("baseDir", getProject().getBasedir().getAbsolutePath());
256            variables.put("buildDir", buildDirectory.getAbsolutePath());
257            variables.put("project.version", getProject().getVersion());
258            variables.put("url", getProject().getUrl());
259            
260            return new MapVariableResolver(variables);
261        }
262    
263        /**
264         * Doc some cleanup and conversion on the Maven project version.
265         * <ul>
266         * <li>-: any - is replaced by +</li>
267         * <li>SNAPSHOT: replace "SNAPSHOT" par of the version with the current time and date</li>
268         * </ul>
269         *
270         * @return the Maven project version
271         */
272        private String getProjectVersion() {
273            String version = getProject().getVersion().replace('-', '+');
274    
275            if (this.timestamped && version.endsWith("+SNAPSHOT")) {
276                version = version.substring(0, version.length() - "+SNAPSHOT".length());
277                version += "~";
278                version += new SimpleDateFormat("yyyyMMdd.HHmmss.SSS").format(new Date());
279            }
280    
281            return version;
282        }
283    
284        /**
285         * @return whether or not Maven is currently operating in the execution root
286         */
287        private boolean isSubmodule() {
288            // FIXME there must be a better way
289            return !session.getExecutionRootDirectory().equalsIgnoreCase(baseDir.toString());
290        }
291    
292        /**
293         * Main entry point
294         *
295         * @throws MojoExecutionException on error
296         */
297        public void execute() throws MojoExecutionException {
298    
299            final MavenProject project = getProject();
300    
301            if(isSubmodule() && !submodules) {
302                getLog().info("skipping sub module: jdeb executing at top-level only");
303                return;
304            }
305    
306            setData(dataSet);
307    
308            try {
309    
310                final VariableResolver resolver = initializeVariableResolver(new HashMap());
311    
312                final File debFile = new File(Utils.replaceVariables(resolver, deb, openReplaceToken, closeReplaceToken));
313                final File controlDirFile = new File(Utils.replaceVariables(resolver, controlDir, openReplaceToken, closeReplaceToken));
314                final File installDirFile = new File(Utils.replaceVariables(resolver, installDir, openReplaceToken, closeReplaceToken));
315                final File changesInFile = new File(Utils.replaceVariables(resolver, changesIn, openReplaceToken, closeReplaceToken));
316                final File changesOutFile = new File(Utils.replaceVariables(resolver, changesOut, openReplaceToken, closeReplaceToken));
317                final File changesSaveFile = new File(Utils.replaceVariables(resolver, changesSave, openReplaceToken, closeReplaceToken));
318    
319                // if there are no producers defined we try to use the artifacts
320                if (dataProducers.isEmpty()) {
321    
322                    final Set<Artifact> artifacts = project.getArtifacts();
323                    if (artifacts.size() == 0) {
324                        // no artifacts
325                        final String packaging = project.getPackaging();
326                        if ("pom".equalsIgnoreCase(packaging)) {
327                            getLog().warn("Creating empty debian package.");
328                        } else {
329                            throw new MojoExecutionException(
330                                "Nothing to include into the debian package. " +
331                                "Did you maybe forget to add a <data> tag or call the plugin directly?");
332                        }
333    
334                    } else {
335    
336                        // attach artifacts (jar, war, etc)
337                        for(Artifact artifact : artifacts) {
338                            final File file = artifact.getFile();
339                            if (file != null) {
340                                dataProducers.add(new DataProducer() {
341                                    public void produce(final DataConsumer receiver) {
342                                        try {
343                                            receiver.onEachFile(
344                                                    new FileInputStream(file),
345                                                    new File(installDirFile, file.getName()).getAbsolutePath(),
346                                                    "",
347                                                    "root", 0, "root", 0,
348                                                    TarEntry.DEFAULT_FILE_MODE,
349                                                    file.length());
350                                        } catch (Exception e) {
351                                            getLog().error(e);
352                                        }
353                                    }
354                                });
355                            } else {
356                                getLog().error("No file for artifact " + artifact);
357                            }
358                        }
359                    }
360                }
361    
362                Console infoConsole = new Console() {
363                    public void info(String s) {
364                        getLog().info(s);
365                    }
366                    public void warn(String s) {
367                        getLog().warn(s);
368                    }
369                };
370    
371                try {
372    
373                    DebMaker debMaker = new DebMaker(infoConsole, debFile, controlDirFile, dataProducers, resolver);
374    
375                    if (changesInFile.exists() && changesInFile.canRead()) {
376                        debMaker.setChangesIn(changesInFile);
377                        debMaker.setChangesOut(changesOutFile);
378                        debMaker.setChangesSave(changesSaveFile);
379                    }
380    
381                    debMaker.setCompression(compression);
382                    debMaker.makeDeb();
383    
384                    // Always attach unless explicitly set to false
385                    if ("true".equalsIgnoreCase(attach)) {
386                        getLog().info("Attaching created debian archive " + debFile);
387                        projectHelper.attachArtifact(project, type, classifier, debFile);
388                    }
389    
390                } catch (PackagingException e) {
391                    getLog().error("Failed to create debian package " + debFile, e);
392                    throw new MojoExecutionException("Failed to create debian package " + debFile, e);
393                }
394    
395            } catch (ParseException e) {
396                throw new MojoExecutionException("Failed parsing pattern", e);
397            }
398        }
