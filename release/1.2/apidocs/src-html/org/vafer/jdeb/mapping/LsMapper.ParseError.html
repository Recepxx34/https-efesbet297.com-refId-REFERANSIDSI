016package org.vafer.jdeb.mapping;
017
018import java.io.BufferedReader;
019import java.io.IOException;
020import java.io.InputStream;
021import java.io.InputStreamReader;
022import java.util.HashMap;
023import java.util.Map;
024import java.util.regex.Matcher;
025import java.util.regex.Pattern;
026
027import org.apache.commons.compress.archivers.tar.TarArchiveEntry;
028
029/**
030 * Reads permissions and ownerships from a "ls -laR > mapping.txt" dump and
031 * maps entries accordingly.
032 *
033 * @author Torsten Curdt
034 */
035public final class LsMapper implements Mapper {
036
037    private final Map<String, TarArchiveEntry> mapping;
038
039
040    public final static class ParseError extends Exception {
041
042        private static final long serialVersionUID = 1L;
043
044        public ParseError( String message ) {
045            super(message);
046        }
047    }
048
049
050    public LsMapper( final InputStream pInput ) throws IOException, ParseError {
051        mapping = parse(pInput);
052    }
053
054    /*
055./trunk/target/test-classes/org/vafer/dependency:
056total 176
057drwxr-xr-x   23 tcurdt  tcurdt   782 Jun 25 03:48 .
058drwxr-xr-x    3 tcurdt  tcurdt   102 Jun 25 03:48 ..
059-rw-r--r--    1 tcurdt  tcurdt  2934 Jun 25 03:48 DependenciesTestCase.class
060-rw-r--r--    1 tcurdt  tcurdt   786 Jun 25 03:48 JarCombiningTestCase$1.class
061-rw-r--r--    1 tcurdt  tcurdt  2176 Jun 25 03:48 WarTestCase.class
062drwxr-xr-x    4 tcurdt  tcurdt   136 Jun 25 03:48 classes
063
064./trunk/target/test-classes/org/vafer/dependency/classes:
065     */
066
067    final private Pattern basePattern = Pattern.compile("^\\./(.*):$");
068    final private Pattern totalPattern = Pattern.compile("^total ([0-9]+)$");
069    final private Pattern dirPattern = Pattern.compile("^d([rwx-]{9})\\s+([0-9]+)\\s+(\\S*)\\s+(\\S*)\\s+([0-9]+)\\s+(.*)\\s+[\\.]{1,2}$");
070    final private Pattern filePattern = Pattern.compile("^([d-])([rwx-]{9})\\s+([0-9]+)\\s+(\\S*)\\s+(\\S*)\\s+([0-9]+)\\s+(.*)\\s+(.*)$");
071    final private Pattern newlinePattern = Pattern.compile("$");
072
073    private String readBase( final BufferedReader reader ) throws IOException, ParseError {
074        final String line = reader.readLine();
075        if (line == null) {
076            return null;
077        }
078        final Matcher matcher = basePattern.matcher(line);
079        if (!matcher.matches()) {
080            throw new ParseError("expected base line but got \"" + line + "\"");
081        }
082        return matcher.group(1);
083    }
084
085    private String readTotal( final BufferedReader reader ) throws IOException, ParseError {
086        final String line = reader.readLine();
087        final Matcher matcher = totalPattern.matcher(line);
088        if (!matcher.matches()) {
089            throw new ParseError("expected total line but got \"" + line + "\"");
090        }
091        return matcher.group(1);
092    }
093
094    private TarArchiveEntry readDir( final BufferedReader reader, final String base ) throws IOException, ParseError {
095        final String current = reader.readLine();
096        final Matcher currentMatcher = dirPattern.matcher(current);
097        if (!currentMatcher.matches()) {
098            throw new ParseError("expected dirline but got \"" + current + "\"");
099        }
100
101        final String parent = reader.readLine();
102        final Matcher parentMatcher = dirPattern.matcher(parent);
103        if (!parentMatcher.matches()) {
104            throw new ParseError("expected dirline but got \"" + parent + "\"");
105        }
106
107        final TarArchiveEntry entry = new TarArchiveEntry(base, true);
108
109        entry.setMode(convertModeFromString(currentMatcher.group(1)));
110        entry.setUserName(currentMatcher.group(3));
111        entry.setGroupName(currentMatcher.group(4));
112
113        return entry;
114    }
115
116
117    private int convertModeFromString( final String mode ) {
118
119        final char[] m = mode.toCharArray();
120        /*
121           -rwxrwxrwx
122
123           4000    set-user-ID-on-execution bit
124           2000    set-user-ID-on-execution bit
125           1000    sticky bit
126           0400    allow read by owner.
127           0200    allow write by owner.
128           0100    execute / search
129           0040    allow read by group members.
130           0020    allow write by group members.
131           0010    execute / search
132           0004    allow read by others.
133           0002    allow write by others.
134           0001    execute / search
135         */
136        // TODO: simplified - needs fixing
137        int sum = 0;
138        int bit = 1;
139        for (int i = m.length - 1; i >= 0; i--) {
140            if (m[i] != '-') {
141                sum += bit;
142            }
143            bit += bit;
144        }
145        return sum;
146    }
147
148    private TarArchiveEntry readFile( final BufferedReader reader, final String base ) throws IOException, ParseError {
149
150        while (true) {
151            final String line = reader.readLine();
152
153            if (line == null) {
154                return null;
155            }
156
157            final Matcher currentMatcher = filePattern.matcher(line);
158            if (!currentMatcher.matches()) {
159                final Matcher newlineMatcher = newlinePattern.matcher(line);
160                if (newlineMatcher.matches()) {
161                    return null;
162                }
163                throw new ParseError("expected file line but got \"" + line + "\"");
164            }
165
166            final String type = currentMatcher.group(1);
167            if (type.startsWith("-")) {
168                final TarArchiveEntry entry = new TarArchiveEntry(base + "/" + currentMatcher.group(8), true);
169
170                entry.setMode(convertModeFromString(currentMatcher.group(2)));
171                entry.setUserName(currentMatcher.group(4));
172                entry.setGroupName(currentMatcher.group(5));
173
174                return entry;
175            }
176        }
177
178    }
179
180    private Map<String, TarArchiveEntry> parse( final InputStream pInput ) throws IOException, ParseError {
181        final Map<String, TarArchiveEntry> mapping = new HashMap<String, TarArchiveEntry>();
182
183        final BufferedReader reader = new BufferedReader(new InputStreamReader(pInput));
184
185        boolean first = true;
186        while (true) {
187
188            final String base;
189            if (first) {
190                base = "";
191                first = false;
192            } else {
193                base = readBase(reader);
194                if (base == null) {
195                    break;
196                }
197            }
198
199            readTotal(reader);
200            final TarArchiveEntry dir = readDir(reader, base);
201            mapping.put(dir.getName(), dir);
202
203            while (true) {
204                final TarArchiveEntry file = readFile(reader, base);
205
206                if (file == null) {
207                    break;
208                }
209
210                mapping.put(file.getName(), file);
211            }
212        }
213
214        return mapping;
215    }
216
217    public TarArchiveEntry map( final TarArchiveEntry pEntry ) {
218
219        final TarArchiveEntry entry = mapping.get(pEntry.getName());
220
221        if (entry != null) {
222
223            final TarArchiveEntry newEntry = new TarArchiveEntry(entry.getName(), true);
224            newEntry.setUserId(entry.getUserId());
225            newEntry.setGroupId(entry.getGroupId());
226            newEntry.setUserName(entry.getUserName());
227            newEntry.setGroupName(entry.getGroupName());
228            newEntry.setMode(entry.getMode());
229            newEntry.setSize(entry.getSize());
230
231            return newEntry;
232        }
233
234        return pEntry;
235    }
