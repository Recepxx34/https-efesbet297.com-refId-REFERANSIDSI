067        private static final class Total {
068            private BigInteger count = BigInteger.valueOf(0);
069    
070            public void add(long size) {
071                count = count.add(BigInteger.valueOf(size));
072            }
073    
074            public String toString() {
075                return "" + count;
076            }
077        }

084        private void addTo( final ArArchiveOutputStream pOutput, final String pName, final String pContent ) throws IOException {
085            final byte[] content = pContent.getBytes();
086            pOutput.putArchiveEntry(new ArArchiveEntry(pName, content.length));
087            pOutput.write(content);
088            pOutput.closeArchiveEntry();
089        }
090    
091        private void addTo( final ArArchiveOutputStream pOutput, final String pName, final File pContent ) throws IOException {
092            pOutput.putArchiveEntry(new ArArchiveEntry(pName, pContent.length()));
093    
094            final InputStream input = new FileInputStream(pContent);
095            try {
096                Utils.copy(input, pOutput);
097            } finally {
098                input.close();
099            }
100    
101            pOutput.closeArchiveEntry();
102        }

104        /**
105         * Create the debian archive with from the provided control files and data producers.
106         *
107         * @param pControlFiles
108         * @param pData
109         * @param pOutput
110         * @param compression the compression method used for the data file (gzip, bzip2 or anything else for no compression)
111         * @return PackageDescriptor
112         * @throws PackagingException
113         */
114        public PackageDescriptor createDeb( final File[] pControlFiles, final DataProducer[] pData, final File pOutput, String compression ) throws PackagingException {
115    
116            File tempData = null;
117            File tempControl = null;
118    
119            try {
120                tempData = File.createTempFile("deb", "data");
121                tempControl = File.createTempFile("deb", "control");
122    
123                console.println("Building data");
124                final StringBuilder md5s = new StringBuilder();
125                final BigInteger size = buildData(pData, tempData, md5s, compression);
126    
127                console.println("Building control");
128                final PackageDescriptor packageDescriptor = buildControl(pControlFiles, size, md5s, tempControl);
129    
130                if (!packageDescriptor.isValid()) {
131                    throw new PackagingException("Control file descriptor keys are invalid " + packageDescriptor.invalidKeys() +
132                        ". The following keys are mandatory " + Arrays.toString(PackageDescriptor.mandatoryKeys) +
133                        ". Please check your pom.xml/build.xml and your control file.");
134                }
135    
136                pOutput.getParentFile().mkdirs();
137    
138                // pass through stream chain to calculate all the different digests
139                final InformationOutputStream md5output = new InformationOutputStream(new FileOutputStream(pOutput), MessageDigest.getInstance("MD5"));
140                final InformationOutputStream sha1output = new InformationOutputStream(md5output, MessageDigest.getInstance("SHA1"));
141                final InformationOutputStream sha256output = new InformationOutputStream(sha1output, MessageDigest.getInstance("SHA-256"));
142                final ArArchiveOutputStream ar = new ArArchiveOutputStream(sha256output);
143    
144                addTo(ar, "debian-binary", "2.0\n");
145                addTo(ar, "control.tar.gz", tempControl);
146                addTo(ar, "data.tar" + getExtension(compression), tempData);
147    
148                ar.close();
149    
150                // intermediate values
151                packageDescriptor.set("MD5", md5output.getHexDigest());
152                packageDescriptor.set("SHA1", sha1output.getHexDigest());
153                packageDescriptor.set("SHA256", sha256output.getHexDigest());
154                packageDescriptor.set("Size", "" + md5output.getSize());
155                packageDescriptor.set("File", pOutput.getName());
156    
157                return packageDescriptor;
158    
159            } catch(Exception e) {
160                throw new PackagingException("Could not create deb package", e);
161            } finally {
162                if (tempData != null) {
163                    if (!tempData.delete()) {
164                        throw new PackagingException("Could not delete " + tempData);
165                    }
166                }
167                if (tempControl != null) {
168                    if (!tempControl.delete()) {
169                        throw new PackagingException("Could not delete " + tempControl);
170                    }
171                }
172            }
173        }

175        /**
176         * Return the extension of a file compressed with the specified method.
177         *
178         * @param pCompression the compression method used
179         * @return
180         */
181        private String getExtension( final String pCompression ) {
182            if ("gzip".equals(pCompression)) {
183                return ".gz";
184            } else if ("bzip2".equals(pCompression)) {
185                return ".bz2";
186            } else {
187                return "";
188            }
189        }

191        /**
192         * Create changes file based on the provided PackageDescriptor.
193         * If pRing, pKey and pPassphrase are provided the changes file will also be signed.
194         * It returns a ChangesDescriptor reflecting the changes
195         * @param pPackageDescriptor
196         * @param pChangesProvider
197         * @param pRing
198         * @param pKey
199         * @param pPassphrase
200         * @param pOutput
201         * @return ChangesDescriptor
202         * @throws IOException
203         * @throws PackagingException
204         */
205        public ChangesDescriptor createChanges( final PackageDescriptor pPackageDescriptor, final ChangesProvider pChangesProvider, final InputStream pRing, final String pKey, final String pPassphrase, final OutputStream pOutput ) throws IOException, PackagingException {
206    
207            final ChangeSet[] changeSets = pChangesProvider.getChangesSets();
208            final ChangesDescriptor changesDescriptor = new ChangesDescriptor(pPackageDescriptor, changeSets);
209    
210            changesDescriptor.set("Format", "1.8");
211    
212            if (changesDescriptor.get("Binary") == null) {
213                changesDescriptor.set("Binary", changesDescriptor.get("Package"));
214            }
215    
216            if (changesDescriptor.get("Source") == null) {
217                changesDescriptor.set("Source", changesDescriptor.get("Package"));
218            }
219    
220            final StringBuilder checksumsSha1 = new StringBuilder("\n");
221            // Checksums-Sha1:
222            // 56ef4c6249dc3567fd2967f809c42d1f9b61adf7 45964 jdeb.deb
223            checksumsSha1.append(' ').append(changesDescriptor.get("SHA1"));
224            checksumsSha1.append(' ').append(changesDescriptor.get("Size"));
225            checksumsSha1.append(' ').append(changesDescriptor.get("File"));
226            changesDescriptor.set("Checksums-Sha1", checksumsSha1.toString());
227    
228            final StringBuilder checksumsSha256 = new StringBuilder("\n");
229            // Checksums-Sha256:
230            // 38c6fa274eb9299a69b739bcbdbd05c7ffd1d8d6472f4245ed732a25c0e5d616 45964 jdeb.deb
231            checksumsSha256.append(' ').append(changesDescriptor.get("SHA256"));
232            checksumsSha256.append(' ').append(changesDescriptor.get("Size"));
233            checksumsSha256.append(' ').append(changesDescriptor.get("File"));
234            changesDescriptor.set("Checksums-Sha256", checksumsSha256.toString());
235    
236    
237            final StringBuilder files = new StringBuilder("\n");
238            files.append(' ').append(changesDescriptor.get("MD5"));
239            files.append(' ').append(changesDescriptor.get("Size"));
240            files.append(' ').append(changesDescriptor.get("Section"));
241            files.append(' ').append(changesDescriptor.get("Priority"));
242            files.append(' ').append(changesDescriptor.get("File"));
243            changesDescriptor.set("Files", files.toString());
244    
245            if (!changesDescriptor.isValid()) {
246                throw new PackagingException("Changes file descriptor keys are invalid " + changesDescriptor.invalidKeys() +
247                    ". The following keys are mandatory " + Arrays.toString(ChangesDescriptor.mandatoryKeys) +
248                    ". Please check your pom.xml/build.xml and your control file.");
249            }
250    
251            final String changes = changesDescriptor.toString();
252            final byte[] changesBytes = changes.getBytes("UTF-8");
253    
254            if (pRing == null || pKey == null || pPassphrase == null) {
255                pOutput.write(changesBytes);
256                pOutput.close();
257                return changesDescriptor;
258            }
259    
260            console.println("Signing changes with key " + pKey);
261    
262            final InputStream input = new ByteArrayInputStream(changesBytes);
263    
264            try {
265                SigningUtils.clearSign(input, pRing, pKey, pPassphrase, pOutput);
266            } catch (Exception e) {
267                e.printStackTrace();
268            }
269    
270            pOutput.close();
271    
272            return changesDescriptor;
273        }

275        /**
276         * Build control archive of the deb
277         * @param pControlFiles
278         * @param pDataSize
279         * @param pChecksums
280         * @param pOutput
281         * @return
282         * @throws FileNotFoundException
283         * @throws IOException
284         * @throws ParseException
285         */
286        private PackageDescriptor buildControl( final File[] pControlFiles, final BigInteger pDataSize, final StringBuilder pChecksums, final File pOutput ) throws IOException, ParseException {
287    
288            final File dir = pOutput.getParentFile();
289            if (dir != null && (!dir.exists() || !dir.isDirectory())) {
290                throw new IOException("Cannot write control file at '" + pOutput.getAbsolutePath() + "'");
291            }
292    
293            final TarOutputStream outputStream = new TarOutputStream(new GZIPOutputStream(new FileOutputStream(pOutput)));
294            outputStream.setLongFileMode(TarOutputStream.LONGFILE_GNU);
295    
296            // create a descriptor out of the "control" file, copy all other files, ignore directories
297            PackageDescriptor packageDescriptor = null;
298            for (int i = 0; i < pControlFiles.length; i++) {
299                final File file = pControlFiles[i];
300    
301                if (file.isDirectory()) {
302                    console.println("Found directory '" + file + "' in the control directory. Maybe you are pointing to wrong dir?");
303                    continue;
304                }
305    
306                final TarEntry entry = new TarEntry(file);
307                final String name = file.getName();
308    
309                entry.setName("./" + name);
310                entry.setNames("root", "root");
311                entry.setMode(PermMapper.toMode("755"));
312    
313                if ("control".equals(name)) {
314    
315                    packageDescriptor = new PackageDescriptor(new FileInputStream(file), resolver);
316    
317                    if (packageDescriptor.get("Date") == null) {
318                        // Mon, 26 Mar 2007 11:44:04 +0200 (RFC 2822)
319                        SimpleDateFormat fmt = new SimpleDateFormat("EEE, d MMM yyyy HH:mm:ss Z", Locale.ENGLISH);
320                        packageDescriptor.set("Date", fmt.format(new Date()));
321                    }
322    
323                    if (packageDescriptor.get("Distribution") == null) {
324                        packageDescriptor.set("Distribution", "unknown");
325                    }
326    
327                    if (packageDescriptor.get("Urgency") == null) {
328                        packageDescriptor.set("Urgency", "low");
329                    }
330    
331                    packageDescriptor.set("Installed-Size", pDataSize.divide(BigInteger.valueOf(1024)).toString());
332    
333                    final String debFullName = System.getenv("DEBFULLNAME");
334                    final String debEmail = System.getenv("DEBEMAIL");
335    
336                    if (debFullName != null && debEmail != null) {
337                        final String maintainer = debFullName + " <" + debEmail + ">";
338                        packageDescriptor.set("Maintainer", maintainer);
339                        console.println("Using maintainer '" + maintainer + "' from the environment variables.");
340                    }
341    
342                } else {
343    
344                    final InformationInputStream infoStream = new InformationInputStream(new FileInputStream(file));
345                    
346                    outputStream.putNextEntry(entry);
347                    Utils.copy(infoStream, outputStream);
348                    outputStream.closeEntry();
349    
350                    if (infoStream.isShell() && !infoStream.hasUnixLineEndings()) {
351                        console.println("WARNING: The file '" + file + "' does not use Unix line endings.");
352                    }
353                }
354            }
355    
356            if (packageDescriptor == null) {
357                throw new FileNotFoundException("No 'control' found in " + Arrays.toString(pControlFiles));
358            }
359    
360            addEntry("control", packageDescriptor.toString(), outputStream);
361            addEntry("md5sums", pChecksums.toString(), outputStream);
362    
363            outputStream.close();
364    
365            return packageDescriptor;
366        }

369        // FIXME temporary - only until Commons Compress is fixed
370        private OutputStream compressedOutputStream( String pCompression, final OutputStream outputStream) throws CompressorException {
371          if ("none".equalsIgnoreCase(pCompression)) {
372            return outputStream;
373          }
374            if ("gzip".equals(pCompression)) {
375              pCompression = "gz";
376            }
377            return new CompressorStreamFactory().createCompressorOutputStream(pCompression, outputStream);
378        }

380        /**
381         * Build the data archive of the deb from the provided DataProducers
382         * @param pData
383         * @param pOutput
384         * @param pChecksums
385         * @param pCompression the compression method used for the data file (gzip, bzip2 or anything else for no compression)
386         * @return
387         * @throws NoSuchAlgorithmException
388         * @throws IOException
389         * @throws CompressorException
390         */
391        BigInteger buildData( final DataProducer[] pData, final File pOutput, final StringBuilder pChecksums, String pCompression ) throws NoSuchAlgorithmException, IOException, CompressorException {
392    
393            final File dir = pOutput.getParentFile();
394            if (dir != null && (!dir.exists() || !dir.isDirectory())) {
395                throw new IOException("Cannot write data file at '" + pOutput.getAbsolutePath() + "'");
396            }
397    
398            final OutputStream fileOutputStream = new FileOutputStream(pOutput);
399            final OutputStream compressedOutputStream = compressedOutputStream(pCompression, fileOutputStream);
400            final TarOutputStream tarOutputStream = new TarOutputStream(compressedOutputStream);
401            tarOutputStream.setLongFileMode(TarOutputStream.LONGFILE_GNU);
402    
403            final MessageDigest digest = MessageDigest.getInstance("MD5");
404    
405            final Total dataSize = new Total();
406    
407            final List addedDirectories = new ArrayList();
408            final DataConsumer receiver = new DataConsumer() {
409                public void onEachDir( String dirname, String linkname, String user, int uid, String group, int gid, int mode, long size ) throws IOException {
410                    dirname = fixPath(dirname);
411    
412                    createParentDirectories((new File(dirname)).getParent(), user, uid, group, gid);
413    
414                    // The directory passed in explicitly by the caller also gets the passed-in mode.  (Unlike
415                    // the parent directories for now.  See related comments at "int mode =" in
416                    // createParentDirectories, including about a possible bug.)
417                    createDirectory(dirname, user, uid, group, gid, mode, 0);
418    
419                    console.println("dir: " + dirname);
420                }
421    
422                public void onEachFile( InputStream inputStream, String filename, String linkname, String user, int uid, String group, int gid, int mode, long size ) throws IOException {
423                    filename = fixPath(filename);
424    
425                    createParentDirectories((new File(filename)).getParent(), user, uid, group, gid);
426    
427                    final TarEntry entry = new TarEntry(filename);
428    
429                    // FIXME: link is in the constructor
430                    entry.setUserName(user);
431                    entry.setUserId(uid);
432                    entry.setGroupName(group);
433                    entry.setGroupId(gid);
434                    entry.setMode(mode);
435                    entry.setSize(size);
436    
437                    tarOutputStream.putNextEntry(entry);
438    
439                    dataSize.add(size);
440                    digest.reset();
441    
442                    Utils.copy(inputStream, new DigestOutputStream(tarOutputStream, digest));
443    
444                    final String md5 = Utils.toHex(digest.digest());
445    
446                    tarOutputStream.closeEntry();
447    
448                    console.println(
449                            "file:" + entry.getName() +
450                            " size:" + entry.getSize() +
451                            " mode:" + entry.getMode() +
452                            " linkname:" + entry.getLinkName() +
453                            " username:" + entry.getUserName() +
454                            " userid:" + entry.getUserId() +
455                            " groupname:" + entry.getGroupName() +
456                            " groupid:" + entry.getGroupId() +
457                            " modtime:" + entry.getModTime() +
458                            " md5: " + md5
459                    );
460    
461                    // append to file md5 list
462                    pChecksums.append(md5).append(" ").append(entry.getName()).append('\n');
463                }

465                private String fixPath(String path) {
466                    // If we're receiving directory names from Windows, then we'll convert to use slash
467                    // This does eliminate the ability to use of a backslash in a directory name on *NIX,
468                  // but in practice, this is a non-issue
469                    if (path.indexOf('\\') > -1) {
470                        path = path.replace('\\', '/');
471                    }
472                    // ensure the path is like : ./foo/bar
473                    if (path.startsWith("/")) {
474                        path = "." + path;
475                    } else if (!path.startsWith("./")) {
476                        path = "./" + path;
477                    }
478                    return path;
479                }

481                private void createDirectory(String directory, String user, int uid, String group, int gid, int mode, long size) throws IOException {
482                    // All dirs should end with "/" when created, or the test DebAndTaskTestCase.testTarFileSet() thinks its a file
483                    // and so thinks it has the wrong permission.
484                    // This consistency also helps when checking if a directory already exists in addedDirectories.
485    
486                    if (!directory.endsWith("/")) {
487                        directory += "/";
488                    }
489    
490                    if (!addedDirectories.contains(directory)) {
491                        TarEntry entry = new TarEntry(directory);
492                        entry.setUserName(user);
493                        entry.setUserId(uid);
494                        entry.setGroupName(group);
495                        entry.setGroupId(gid);
496                        entry.setMode(mode);
497                        entry.setSize(size);
498    
499                        tarOutputStream.putNextEntry(entry);
500                        tarOutputStream.closeEntry();
501                        addedDirectories.add(directory); // so addedDirectories consistently have "/" for finding duplicates.
502                    }
503                }
504    
505                private void createParentDirectories(String dirname, String user, int uid, String group, int gid) throws IOException {
506                    // Debian packages must have parent directories created
507                    // before sub-directories or files can be installed.
508                    // For example, if an entry of ./usr/lib/foo/bar existed
509                    // in a .deb package, but the ./usr/lib/foo directory didn't
510                    // exist, the package installation would fail.  The .deb must
511                    // then have an entry for ./usr/lib/foo and then ./usr/lib/foo/bar
512    
513                    if (dirname == null) {
514                      return;
515                    }
516    
517                    // The loop below will create entries for all parent directories
518                    // to ensure that .deb packages will install correctly.
519                    String[] pathParts = dirname.split("\\/");
520                    String parentDir = "./";
521                    for (int i = 1; i < pathParts.length; i++) {
522                        parentDir += pathParts[i] + "/";
523                        // Make it so the dirs can be traversed by users.
524                        // We could instead try something more granular, like setting the directory
525                        // permission to 'rx' for each of the 3 user/group/other read permissions
526                        // found on the file being added (ie, only if "other" has read
527                        // permission on the main node, then add o+rx permission on all the containing
528                        // directories, same w/ user & group), and then also we'd have to
529                        // check the parentDirs collection of those already added to
530                        // see if those permissions need to be similarly updated.  (Note, it hasn't
531                        // been demonstrated, but there might be a bug if a user specifically
532                        // requests a directory with certain permissions,
533                        // that has already been auto-created because it was a parent, and if so, go set
534                        // the user-requested mode on that directory instead of this automatic one.)
535                        // But for now, keeping it simple by making every dir a+rx.   Examples are:
536                        // drw-r----- fs/fs   # what you get with setMode(mode)
537                        // drwxr-xr-x fs/fs   # Usable. Too loose?
538                        int mode = TarEntry.DEFAULT_DIR_MODE;
539    
540                        createDirectory(parentDir, user, uid, group, gid, mode, 0);
541                    }
542                }
543            };
544    
545            for (int i = 0; i < pData.length; i++) {
546                final DataProducer data = pData[i];
547                data.produce(receiver);
548            }
549    
550            tarOutputStream.close();
551    
552            console.println("Total size: " + dataSize);
553    
554            return dataSize.count;
555        }

557        private static void addEntry( final String pName, final String pContent, final TarOutputStream pOutput ) throws IOException {
558            final byte[] data = pContent.getBytes("UTF-8");
559    
560            final TarEntry entry = new TarEntry("./" + pName);
561            entry.setSize(data.length);
562            entry.setNames("root", "root");
563    
564            pOutput.putNextEntry(entry);
565            pOutput.write(data);
566            pOutput.closeEntry();
567        }
