016    package org.vafer.jdeb.maven;
017    
018    import java.io.File;
019    import java.io.FileNotFoundException;
020    import java.io.IOException;
021    import java.util.ArrayList;
022    import java.util.List;
023    import java.util.StringTokenizer;
024    
025    import org.vafer.jdeb.DataConsumer;
026    import org.vafer.jdeb.DataProducer;
027    import org.vafer.jdeb.producers.DataProducerArchive;
028    import org.vafer.jdeb.producers.DataProducerDirectory;
029    import org.vafer.jdeb.producers.DataProducerFile;
030    import org.vafer.jdeb.producers.DataProducerPathTemplate;
031    
032    import static org.vafer.jdeb.maven.MissingSourceBehavior.FAIL;
033    import static org.vafer.jdeb.maven.MissingSourceBehavior.IGNORE;
034    
035    /**
036     * Maven "data" element acting as a factory for DataProducers. So far Archive and
037     * Directory producers are supported. Both support the usual ant pattern set
038     * matching.
039     *
040     * @author Bryan Sant
041     */
042    public final class Data implements DataProducer {
043    
044        private File src;
045    
046        /**
047         * @parameter expression="${src}"
048         * @required
049         */
050        public void setSrc( File src ) {
051            this.src = src;
052        }
053    
054        private String dst;
055    
056        /**
057         * @parameter expression="${dst}"
058         * @required
059         */
060        public void setDst( String dst ) {
061            this.dst = dst;
062        }
063    
064        private String type;
065    
066        /**
067         * @parameter expression="${type}"
068         */
069        public void setType( String type ) {
070            this.type = type;
071        }
072    
073        private MissingSourceBehavior missingSrc = FAIL;
074    
075        /**
076         * @parameter expression="${missingSrc}"
077         */
078        public void setMissingSrc( String missingSrc ) {
079            MissingSourceBehavior value = MissingSourceBehavior.valueOf(missingSrc.trim().toUpperCase());
080            if (value == null) {
081                throw new IllegalArgumentException("Unknown " + MissingSourceBehavior.class.getSimpleName() + ": " + missingSrc);
082            }
083            this.missingSrc = value;
084        }
085    
086        /**
087         * @parameter expression="${includes}" alias="includes"
088         */
089        public void setIncludes( String includes ) {
090            includePatterns = splitPatterns(includes);
091        }
092    
093        private String[] includePatterns;
094    
095        /**
096         * @parameter expression="${excludes}" alias="excludes"
097         */
098        public void setExcludes( String excludes ) {
099            excludePatterns = splitPatterns(excludes);
100        }
101    
102        private String[] excludePatterns;
103        /**
104         * @parameter expression="${mapper}"
105         */
106        private Mapper mapper;
107    
108        /**
109         * @parameter expression="${paths}"
110         */
111        private String[] paths;
112    
113        /* For testing only */
114        void setPaths( String[] paths ) {
115            this.paths = paths;
116        }
117    
118        public String[] splitPatterns( String patterns ) {
119            String[] result = null;
120            if (patterns != null && patterns.length() > 0) {
121                List tokens = new ArrayList();
122                StringTokenizer tok = new StringTokenizer(patterns, ", ", false);
123                while (tok.hasMoreTokens()) {
124                    tokens.add(tok.nextToken());
125                }
126                result = (String[]) tokens.toArray(new String[tokens.size()]);
127            }
128            return result;
129        }
130    
131        public void produce( final DataConsumer pReceiver ) throws IOException {
132            if (src != null && !src.exists()) {
133                if (missingSrc == IGNORE) {
134                    return;
135                } else {
136                    throw new FileNotFoundException("Data source not found : " + src);
137                }
138            }
139    
140            if (src == null && (paths == null || paths.length == 0)) {
141                throw new RuntimeException("src or paths not set");
142            }
143    
144            org.vafer.jdeb.mapping.Mapper[] mappers = null;
145            if (mapper != null) {
146                mappers = new org.vafer.jdeb.mapping.Mapper[] { mapper.createMapper() };
147            }
148    
149            if ("file".equalsIgnoreCase(type)) {
150                new DataProducerFile(src, dst, includePatterns, excludePatterns, mappers).produce(pReceiver);
151                return;
152            }
153    
154            if ("archive".equalsIgnoreCase(type)) {
155                new DataProducerArchive(src, includePatterns, excludePatterns, mappers).produce(pReceiver);
156                return;
157            }
158    
159            if ("directory".equalsIgnoreCase(type)) {
160                new DataProducerDirectory(src, includePatterns, excludePatterns, mappers).produce(pReceiver);
161                return;
162            }
163    
164            if ("template".equalsIgnoreCase(type)) {
165                new DataProducerPathTemplate(paths, includePatterns, excludePatterns, mappers).produce(pReceiver);
166                return;
167            }
168    
169            throw new IOException("Unknown type '" + type + "' (file|directory|archive|template) for " + src);
170        }
171    
172    }
