188     private boolean isWritableFile(File file) {
189         return !file.exists() || file.isFile() && file.canWrite();
190     }

193      * Validates the input parameters.
194      */
195     public void validate() throws PackagingException {
196         if (control == null || !control.isDirectory()) {
197             throw new PackagingException("The 'control' attribute doesn't point to a directory.");
198         }
199 
200         if (changesIn != null) {
201 
202             if (changesIn.exists() && (!changesIn.isFile() || !changesIn.canRead())) {
203                 throw new PackagingException("The 'changesIn' setting needs to point to a readable file. " + changesIn + " was not found/readable.");
204             }
205             
206             if (changesOut != null && !isWritableFile(changesOut)) {
207                 throw new PackagingException("Cannot write the output for 'changesOut' to " + changesOut);
208             }
209 
210             if (changesSave != null && !isWritableFile(changesSave)) {
211                 throw new PackagingException("Cannot write the output for 'changesSave' to " + changesSave);
212             }
213 
214         } else {
215             if (changesOut != null || changesSave != null) {
216                 throw new PackagingException("The 'changesOut' or 'changesSave' settings may only be used when there is a 'changesIn' specified.");
217             }
218         }
219 
220         if (Compression.toEnum(compression) == null) {
221             throw new PackagingException("The compression method '" + compression + "' is not supported (expected 'none', 'gzip', 'bzip2' or 'xz')");
222         }
223 
224         if (deb == null) {
225             throw new PackagingException("You need to specify where the deb file is supposed to be created.");
226         }
227         
228         if (dataProducers.size() == 0) {
229             throw new PackagingException("You need to provide at least one reference to a tgz or directory with data.");
230         }
231     }

233     public void makeDeb() throws PackagingException {
234         BinaryPackageControlFile packageControlFile;
235         try {
236             console.info("Creating debian package: " + deb);
237 
238             // If we should sign the package
239             boolean doSign = signPackage;
240 
241             if (keyring == null || !keyring.exists()) {
242                 doSign = false;
243                 console.warn("Signing requested, but no keyring supplied");
244             }
245 
246             if (key == null) {
247                 doSign = false;
248                 console.warn("Signing requested, but no key supplied");
249             }
250 
251             if (passphrase == null) {
252                 doSign = false;
253                 console.warn("Signing requested, but no passphrase supplied");
254             }
255 
256             if (doSign) {
257                 FileInputStream keyRingInput = new FileInputStream(keyring);
258                 PGPSigner signer = null;
259                 try {
260                     signer = new PGPSigner(new FileInputStream(keyring), key, passphrase);
261                 } finally {
262                     keyRingInput.close();
263                 }
264 
265                 int digest = PGPUtil.SHA1;
266 
267                 PGPSignatureGenerator signatureGenerator = new PGPSignatureGenerator(new BcPGPContentSignerBuilder(signer.getSecretKey().getPublicKey().getAlgorithm(), digest));
268                 signatureGenerator.init(PGPSignature.BINARY_DOCUMENT, signer.getPrivateKey());
269 
270                 packageControlFile = createSignedDeb(Compression.toEnum(compression), signatureGenerator);
271             } else {
272                 packageControlFile = createDeb(Compression.toEnum(compression));
273             }
274 
275         } catch (Exception e) {
276             throw new PackagingException("Failed to create debian package " + deb, e);
277         }
278         
279         makeChangesFiles(packageControlFile);
280     }

282     private void makeChangesFiles(final BinaryPackageControlFile packageControlFile) throws PackagingException {
283         if (changesOut == null) {
284             changesOut = new File(deb.getParentFile(), deb.getName().replace(".deb", ".changes"));
285         }
286         
287         ChangesProvider changesProvider;
288         FileOutputStream out = null;
289         
290         try {
291             console.info("Creating changes file: " + changesOut);
292             
293             out = new FileOutputStream(changesOut);
294             
295             if (changesIn != null && changesIn.exists()) {
296                 // read the changes form a textfile provider
297                 changesProvider = new TextfileChangesProvider(new FileInputStream(changesIn), packageControlFile);
298             } else {
299                 // create an empty changelog
300                 changesProvider = new ChangesProvider() {
301                     @Override
302                     public ChangeSet[] getChangesSets() {
303                         return new ChangeSet[] {
304                                 new ChangeSet(packageControlFile.get("Package"),
305                                         packageControlFile.get("Version"),
306                                         new Date(),
307                                         packageControlFile.get("Distribution"),
308                                         packageControlFile.get("Urgency"),
309                                         packageControlFile.get("Maintainer"),
310                                         new String[0])
311                         };
312                     }
313                 };
314             }
315             
316             ChangesFileBuilder builder = new ChangesFileBuilder();
317             ChangesFile changesFile = builder.createChanges(packageControlFile, deb, changesProvider);
318             
319             if (keyring != null && key != null && passphrase != null) {
320                 console.info("Signing the changes file with the key " + key);
321                 PGPSigner signer = new PGPSigner(new FileInputStream(keyring), key, passphrase);
322                 signer.clearSign(changesFile.toString(), out);
323             } else {
324                 out.write(changesFile.toString().getBytes("UTF-8"));
325             }
326             out.flush();
327 
328         } catch (Exception e) {
329             throw new PackagingException("Failed to create the Debian changes file " + changesOut, e);
330         } finally {
331             IOUtils.closeQuietly(out);
332         }
333         
334         if (changesSave == null || !(changesProvider instanceof TextfileChangesProvider)) {
335             return;
336         }
337         
338         try {
339             console.info("Saving changes to file: " + changesSave);
340 
341             ((TextfileChangesProvider) changesProvider).save(new FileOutputStream(changesSave));
342 
343         } catch (Exception e) {
344             throw new PackagingException("Failed to save debian changes file " + changesSave, e);
345         }
346     }

348     private List<String> populateConffiles(Collection<DataProducer> producers) {
349         final List<String> result = new ArrayList<String>();
350         
351         if (producers == null || producers.isEmpty()) {
352             return result;
353         }
354         
355         final DataConsumer receiver = new DataConsumer() {
356             public void onEachDir( String dirname, String linkname, String user, int uid, String group, int gid, int mode, long size ) throws IOException {
357                 //
358             }
359             
360             public void onEachFile( InputStream inputStream, String filename, String linkname, String user, int uid, String group, int gid, int mode, long size ) throws IOException {
361                 String tempConffileItem = filename;
362                 if (tempConffileItem.startsWith(".")) {
363                     tempConffileItem = tempConffileItem.substring(1);
364                 }
365                 result.add(tempConffileItem);
366             }
367 
368             public void onEachLink(String path, String linkname, boolean symlink, String user, int uid, String group, int gid, int mode) throws IOException {
369                 //
370             }
371         };
372 
373         try {
374             for (DataProducer data : producers) {
375                 data.produce(receiver);
376             }
377         } catch(Exception e) {
378             //
379         }
380         
381         return result;
382     }

397     /**
398      * Create the debian archive with from the provided control files and data producers.
399      *
400      * @param compression   the compression method used for the data file (gzip, bzip2 or anything else for no compression)
401      * @param signatureGenerator   the signature generator
402      * 
403      * @return PackageDescriptor
404      * @throws PackagingException
405      */
406     public BinaryPackageControlFile createSignedDeb(Compression compression, final PGPSignatureGenerator signatureGenerator ) throws PackagingException {
407         File tempData = null;
408         File tempControl = null;
409 
410         try {
411             tempData = File.createTempFile("deb", "data");
412             tempControl = File.createTempFile("deb", "control");
413 
414             console.debug("Building data");
415             DataBuilder dataBuilder = new DataBuilder(console);
416             StringBuilder md5s = new StringBuilder();
417             BigInteger size = dataBuilder.buildData(dataProducers, tempData, md5s, compression);
418 
419             console.info("Building conffiles");
420             List<String> tempConffiles = populateConffiles(conffilesProducers);
421             
422             console.debug("Building control");
423             ControlBuilder controlBuilder = new ControlBuilder(console, variableResolver, openReplaceToken, closeReplaceToken);
424             BinaryPackageControlFile packageControlFile = controlBuilder.createPackageControlFile(new File(control, "control"), size);
425             if (packageControlFile.get("Package") == null) {
426                 packageControlFile.set("Package", packageName);
427             }
428             if (packageControlFile.get("Depends") == null) {
429                 packageControlFile.set("Depends", depends);
430             }
431             if (packageControlFile.get("Section") == null) {
432                 packageControlFile.set("Section", section);
433             }
434             if (packageControlFile.get("Description") == null) {
435                 packageControlFile.set("Description", description);
436             }
437             if (packageControlFile.get("Homepage") == null) {
438                 packageControlFile.set("Homepage", homepage);
439             }
440             
441             controlBuilder.buildControl(packageControlFile, control.listFiles(), tempConffiles , md5s, tempControl);
442             
443             if (!packageControlFile.isValid()) {
444                 throw new PackagingException("Control file fields are invalid " + packageControlFile.invalidFields() +
445                     ". The following fields are mandatory: " + packageControlFile.getMandatoryFields() +
446                     ". Please check your pom.xml/build.xml and your control file.");
447             }
448 
449             deb.getParentFile().mkdirs();
450             
451             
452             ArArchiveOutputStream ar = new ArArchiveOutputStream(new FileOutputStream(deb));
453             
454             addTo(ar, "debian-binary", "2.0\n");
455             addTo(ar, "control.tar.gz", tempControl);
456             addTo(ar, "data.tar" + compression.getExtension(), tempData);
457 
458             if (signatureGenerator != null) {
459                 console.info("Signing package with key " + key);
460                 PGPSignatureOutputStream sigStream = new PGPSignatureOutputStream(signatureGenerator);
461                 addTo(sigStream, "2.0\n");
462                 addTo(sigStream, tempControl);
463                 addTo(sigStream, tempData);
464                 addTo(ar, "_gpgorigin", sigStream.generateASCIISignature());
465             }
466 
467             ar.close();
468             
469             return packageControlFile;
470 
471         } catch (Exception e) {
472             throw new PackagingException("Could not create deb package", e);
473         } finally {
474             if (tempData != null) {
475                 if (!tempData.delete()) {
476                     console.warn("Could not delete the temporary file " + tempData);
477                 }
478             }
479             if (tempControl != null) {
480                 if (!tempControl.delete()) {
481                     console.warn("Could not delete the temporary file " + tempControl);
482                 }
483             }
484         }
485     }

487     private void addTo(ArArchiveOutputStream pOutput, String pName, String pContent) throws IOException {
488         final byte[] content = pContent.getBytes();
489         pOutput.putArchiveEntry(new ArArchiveEntry(pName, content.length));
490         pOutput.write(content);
491         pOutput.closeArchiveEntry();
