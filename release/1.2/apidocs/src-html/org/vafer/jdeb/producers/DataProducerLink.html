016package org.vafer.jdeb.producers;
017
018import java.io.IOException;
019
020import org.apache.commons.compress.archivers.tar.TarArchiveEntry;
021import org.vafer.jdeb.DataConsumer;
022import org.vafer.jdeb.DataProducer;
023import org.vafer.jdeb.mapping.Mapper;
024
025/**
026 * DataProducer representing a single file
027 * For cross-platform permissions and ownerships you probably want to use a Mapper, too.
028 *
029 * @author Thomas Mortagne
030 */
031public final class DataProducerLink extends AbstractDataProducer implements DataProducer {
032
033    private final String path;
034    private final String linkName;
035    private final boolean symlink;
036
037    public DataProducerLink(final String path, final String linkName, final boolean symlink, String[] pIncludes, String[] pExcludes, Mapper[] pMapper) {
038        super(pIncludes, pExcludes, pMapper);
039        this.path = path;
040        this.symlink = symlink;
041        this.linkName = linkName;
042    }
043
044    public void produce( final DataConsumer pReceiver ) throws IOException {
045        TarArchiveEntry entry = new TarArchiveEntry(path, symlink ? TarArchiveEntry.LF_SYMLINK : TarArchiveEntry.LF_LINK);
046        entry.setLinkName(linkName);
047
048        entry.setUserId(Producers.ROOT_UID);
049        entry.setUserName(Producers.ROOT_NAME);
050        entry.setGroupId(Producers.ROOT_UID);
051        entry.setGroupName(Producers.ROOT_NAME);
052        entry.setMode(TarArchiveEntry.DEFAULT_FILE_MODE);
053
054        entry = map(entry);
055
056        pReceiver.onEachLink(path, linkName, symlink, entry.getUserName(), entry.getUserId(), entry.getGroupName(), entry.getGroupId(), entry.getMode());
057    }
058
059}
