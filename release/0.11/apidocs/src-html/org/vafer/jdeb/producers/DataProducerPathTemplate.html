016    package org.vafer.jdeb.producers;
017    
018    import java.io.IOException;
019    
020    import org.apache.tools.tar.TarEntry;
021    import org.vafer.jdeb.DataConsumer;
022    import org.vafer.jdeb.DataProducer;
023    import org.vafer.jdeb.mapping.Mapper;
024    
025    public class DataProducerPathTemplate extends AbstractDataProducer implements DataProducer {
026    
027        private final String[] literalPaths;
028    
029        public DataProducerPathTemplate( String[] pLiteralPaths, String[] pIncludes, String[] pExcludes, Mapper[] pMapper ) {
030            super(pIncludes, pExcludes, pMapper);
031            literalPaths = pLiteralPaths;
032        }
033    
034        public void produce( DataConsumer pReceiver ) throws IOException {
035            for (String literalPath : literalPaths) {
036                TarEntry entry = new TarEntry(literalPath);
037                entry.setUserId(0);
038                entry.setUserName("root");
039                entry.setGroupId(0);
040                entry.setGroupName("root");
041                entry.setMode(TarEntry.DEFAULT_DIR_MODE);
042    
043                entry = map(entry);
044    
045                entry.setSize(0);
046    
047                pReceiver.onEachDir(entry.getName(), entry.getLinkName(), entry.getUserName(), entry.getUserId(), entry.getGroupName(), entry.getGroupId(), entry.getMode(), entry.getSize());
048            }
049        }
050    
051    }
