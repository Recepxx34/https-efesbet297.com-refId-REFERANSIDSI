096    private void addTo( final ArArchiveOutputStream pOutput, final String pName, final String pContent ) throws IOException {
097        final byte[] content = pContent.getBytes();
098        pOutput.putArchiveEntry(new ArArchiveEntry(pName, content.length));
099        pOutput.write(content);
100        pOutput.closeArchiveEntry();
101    }
102
103    private void addTo( final ArArchiveOutputStream pOutput, final String pName, final File pContent ) throws IOException {
104        pOutput.putArchiveEntry(new ArArchiveEntry(pName, pContent.length()));
105
106        final InputStream input = new FileInputStream(pContent);
107        try {
108            Utils.copy(input, pOutput);
109        } finally {
110            input.close();
111        }
112
113        pOutput.closeArchiveEntry();
114    }

116    /**
117     * Create the debian archive with from the provided control files and data producers.
118     *
119     * @param pControlFiles
120     * @param pData
121     * @param pOutput
122     * @param compression   the compression method used for the data file (gzip, bzip2 or anything else for no compression)
123     * @return PackageDescriptor
124     * @throws PackagingException
125     */
126    public PackageDescriptor createDeb( final File[] pControlFiles, final DataProducer[] pData, final File pOutput, String compression ) throws PackagingException {
127
128        File tempData = null;
129        File tempControl = null;
130
131        try {
132            tempData = File.createTempFile("deb", "data");
133            tempControl = File.createTempFile("deb", "control");
134
135            console.info("Building data");
136            final StringBuilder md5s = new StringBuilder();
137            final BigInteger size = buildData(pData, tempData, md5s, compression);
138
139            console.info("Building control");
140            final PackageDescriptor packageDescriptor = buildControl(pControlFiles, size, md5s, tempControl);
141
142            if (!packageDescriptor.isValid()) {
143                throw new PackagingException("Control file descriptor keys are invalid " + packageDescriptor.invalidKeys() +
144                    ". The following keys are mandatory " + Arrays.toString(PackageDescriptor.mandatoryKeys) +
145                    ". Please check your pom.xml/build.xml and your control file.");
146            }
147
148            pOutput.getParentFile().mkdirs();
149
150            // pass through stream chain to calculate all the different digests
151            final InformationOutputStream md5output = new InformationOutputStream(new FileOutputStream(pOutput), MessageDigest.getInstance("MD5"));
152            final InformationOutputStream sha1output = new InformationOutputStream(md5output, MessageDigest.getInstance("SHA1"));
153            final InformationOutputStream sha256output = new InformationOutputStream(sha1output, MessageDigest.getInstance("SHA-256"));
154            final ArArchiveOutputStream ar = new ArArchiveOutputStream(sha256output);
155
156            addTo(ar, "debian-binary", "2.0\n");
157            addTo(ar, "control.tar.gz", tempControl);
158            addTo(ar, "data.tar" + getExtension(compression), tempData);
159
160            ar.close();
161
162            // intermediate values
163            packageDescriptor.set("MD5", md5output.getHexDigest());
164            packageDescriptor.set("SHA1", sha1output.getHexDigest());
165            packageDescriptor.set("SHA256", sha256output.getHexDigest());
166            packageDescriptor.set("Size", "" + md5output.getSize());
167            packageDescriptor.set("File", pOutput.getName());
168
169            return packageDescriptor;
170
171        } catch (Exception e) {
172            throw new PackagingException("Could not create deb package", e);
173        } finally {
174            if (tempData != null) {
175                if (!tempData.delete()) {
176                    throw new PackagingException("Could not delete " + tempData);
177                }
178            }
179            if (tempControl != null) {
180                if (!tempControl.delete()) {
181                    throw new PackagingException("Could not delete " + tempControl);
182                }
183            }
184        }
185    }

187    /**
188     * Return the extension of a file compressed with the specified method.
189     *
190     * @param pCompression the compression method used
191     * @return
192     */
193    private String getExtension( final String pCompression ) {
194        if ("gzip".equals(pCompression)) {
195            return ".gz";
196        } else if ("bzip2".equals(pCompression)) {
197            return ".bz2";
198        } else {
199            return "";
200        }

203    /**
204     * Create changes file based on the provided PackageDescriptor.
205     * If pRing, pKey and pPassphrase are provided the changes file will also be signed.
206     * It returns a ChangesDescriptor reflecting the changes
207     *
208     * @param pPackageDescriptor
209     * @param pChangesProvider
210     * @param pRing
211     * @param pKey
212     * @param pPassphrase
213     * @param pOutput
214     * @return ChangesDescriptor
215     * @throws IOException
216     * @throws PackagingException
217     */
218    public ChangesDescriptor createChanges( final PackageDescriptor pPackageDescriptor, final ChangesProvider pChangesProvider, final InputStream pRing, final String pKey, final String pPassphrase, final OutputStream pOutput ) throws IOException, PackagingException {
219
220        final ChangeSet[] changeSets = pChangesProvider.getChangesSets();
221        final ChangesDescriptor changesDescriptor = new ChangesDescriptor(pPackageDescriptor, changeSets);
222
223        changesDescriptor.set("Format", "1.8");
224
225        if (changesDescriptor.get("Binary") == null) {
226            changesDescriptor.set("Binary", changesDescriptor.get("Package"));
227        }
228
229        if (changesDescriptor.get("Source") == null) {
230            changesDescriptor.set("Source", changesDescriptor.get("Package"));
231        }
232
233        final StringBuilder checksumsSha1 = new StringBuilder("\n");
234        // Checksums-Sha1:
235        // 56ef4c6249dc3567fd2967f809c42d1f9b61adf7 45964 jdeb.deb
236        checksumsSha1.append(' ').append(changesDescriptor.get("SHA1"));
237        checksumsSha1.append(' ').append(changesDescriptor.get("Size"));
238        checksumsSha1.append(' ').append(changesDescriptor.get("File"));
239        changesDescriptor.set("Checksums-Sha1", checksumsSha1.toString());
240
241        final StringBuilder checksumsSha256 = new StringBuilder("\n");
242        // Checksums-Sha256:
243        // 38c6fa274eb9299a69b739bcbdbd05c7ffd1d8d6472f4245ed732a25c0e5d616 45964 jdeb.deb
244        checksumsSha256.append(' ').append(changesDescriptor.get("SHA256"));
245        checksumsSha256.append(' ').append(changesDescriptor.get("Size"));
246        checksumsSha256.append(' ').append(changesDescriptor.get("File"));
247        changesDescriptor.set("Checksums-Sha256", checksumsSha256.toString());
248
249
250        final StringBuilder files = new StringBuilder("\n");
251        files.append(' ').append(changesDescriptor.get("MD5"));
252        files.append(' ').append(changesDescriptor.get("Size"));
253        files.append(' ').append(changesDescriptor.get("Section"));
254        files.append(' ').append(changesDescriptor.get("Priority"));
255        files.append(' ').append(changesDescriptor.get("File"));
256        changesDescriptor.set("Files", files.toString());
257
258        if (!changesDescriptor.isValid()) {
259            throw new PackagingException("Changes file descriptor keys are invalid " + changesDescriptor.invalidKeys() +
260                ". The following keys are mandatory " + Arrays.toString(ChangesDescriptor.mandatoryKeys) +
261                ". Please check your pom.xml/build.xml and your control file.");
262        }
263
264        final String changes = changesDescriptor.toString();
265        final byte[] changesBytes = changes.getBytes("UTF-8");
266
267        if (pRing == null || pKey == null || pPassphrase == null) {
268            pOutput.write(changesBytes);
269            pOutput.close();
270            return changesDescriptor;
271        }
272
273        console.info("Signing changes with key " + pKey);
274
275        final InputStream input = new ByteArrayInputStream(changesBytes);
276
277        try {
278            SigningUtils.clearSign(input, pRing, pKey, pPassphrase, pOutput);
279        } catch (Exception e) {
280            e.printStackTrace();
281        }
282
283        pOutput.close();
284
285        return changesDescriptor;
286    }

288    /**
289     * Build control archive of the deb
290     *
291     * @param pControlFiles
292     * @param pDataSize
293     * @param pChecksums
294     * @param pOutput
295     * @return
296     * @throws FileNotFoundException
297     * @throws IOException
298     * @throws ParseException
299     */
300    private PackageDescriptor buildControl( final File[] pControlFiles, final BigInteger pDataSize, final StringBuilder pChecksums, final File pOutput ) throws IOException, ParseException {
301
302        final File dir = pOutput.getParentFile();
303        if (dir != null && (!dir.exists() || !dir.isDirectory())) {
304            throw new IOException("Cannot write control file at '" + pOutput.getAbsolutePath() + "'");
305        }
306
307        final TarArchiveOutputStream outputStream = new TarArchiveOutputStream(new GZIPOutputStream(new FileOutputStream(pOutput)));
308        outputStream.setLongFileMode(TarArchiveOutputStream.LONGFILE_GNU);
309
310
311        // create a descriptor out of the "control" file, copy all other files, ignore directories
312        PackageDescriptor packageDescriptor = null;
313        for (File file : pControlFiles) {
314            if (file.isDirectory()) {
315                // warn about the misplaced directory, except for directories ignored by default (.svn, cvs, etc)
316                boolean isDefaultExcludes = false;
317                for (String pattern : DirectoryScanner.getDefaultExcludes()) {
318                    isDefaultExcludes = DirectoryScanner.match(pattern, file.getAbsolutePath().replace("\\", "/"));
319                    if (isDefaultExcludes) {
320                        break;
321                    }
322                }
323
324                if (!isDefaultExcludes) {
325                    console.info("Found directory '" + file + "' in the control directory. Maybe you are pointing to wrong dir?");
326                }
327                continue;
328            }
329
330            final TarArchiveEntry entry = new TarArchiveEntry(file);
331            final String name = file.getName();
332
333            entry.setName("./" + name);
334            entry.setNames("root", "root");
335            entry.setMode(PermMapper.toMode("755"));
336
337            if (CONFIGURATION_FILENAMES.contains(name)) {
338
339                FilteredConfigurationFile configurationFile = new FilteredConfigurationFile(file.getName(), new FileInputStream(file), resolver);
340                configurationFiles.add(configurationFile);
341
342            } else if ("control".equals(name)) {
343
344                packageDescriptor = new PackageDescriptor(new FileInputStream(file), resolver);
345
346                if (packageDescriptor.get("Date") == null) {
347                    // Mon, 26 Mar 2007 11:44:04 +0200 (RFC 2822)
348                    SimpleDateFormat fmt = new SimpleDateFormat("EEE, d MMM yyyy HH:mm:ss Z", Locale.ENGLISH);
349                    packageDescriptor.set("Date", fmt.format(new Date()));
350                }
351
352                if (packageDescriptor.get("Distribution") == null) {
353                    packageDescriptor.set("Distribution", "unknown");
354                }
355
356                if (packageDescriptor.get("Urgency") == null) {
357                    packageDescriptor.set("Urgency", "low");
358                }
359
360                packageDescriptor.set("Installed-Size", pDataSize.divide(BigInteger.valueOf(1024)).toString());
361
362                final String debFullName = System.getenv("DEBFULLNAME");
363                final String debEmail = System.getenv("DEBEMAIL");
364
365                if (debFullName != null && debEmail != null) {
366                    final String maintainer = debFullName + " <" + debEmail + ">";
367                    packageDescriptor.set("Maintainer", maintainer);
368                    console.info("Using maintainer '" + maintainer + "' from the environment variables.");
369                }
370
371            } else {
372
373                InformationInputStream infoStream = new InformationInputStream(new FileInputStream(file));
374                Utils.copy(infoStream, NullOutputStream.NULL_OUTPUT_STREAM);
375                infoStream.close();
376
377                InputStream in = new FileInputStream(file);
378                if (infoStream.isShell()) {
379                    if(!infoStream.hasUnixLineEndings()) {
380                        // fix the line endings automatically
381                        byte[] buf = Utils.toUnixLineEndings(in);
382                        entry.setSize(buf.length);
383                        in = new ByteArrayInputStream(buf);
384                    }
385                } else {
386                    // entry.setMode(PermMapper.toMode("644"));
387                }
388
389                outputStream.putArchiveEntry(entry);
390                Utils.copy(in, outputStream);
391                outputStream.closeArchiveEntry();
392                in.close();
393            }
394        }
395
396        if (packageDescriptor == null) {
397            throw new FileNotFoundException("No 'control' found in " + Arrays.toString(pControlFiles));
398        }
399
400        for (FilteredConfigurationFile configurationFile : configurationFiles) {
401            addControlEntry(configurationFile.getName(), configurationFile.toString(), outputStream);
402        }
403        addEntry("control", packageDescriptor.toString(), outputStream);
404        addEntry("md5sums", pChecksums.toString(), outputStream);
405
406        outputStream.close();
407
408        return packageDescriptor;
409    }

412    // FIXME temporary - only until Commons Compress is fixed
413    private OutputStream compressedOutputStream( String pCompression, final OutputStream outputStream ) throws CompressorException {
414        if ("none".equalsIgnoreCase(pCompression)) {
415            return outputStream;
416        }
417        if ("gzip".equals(pCompression)) {
418            pCompression = "gz";
419        }
420        return new CompressorStreamFactory().createCompressorOutputStream(pCompression, outputStream);
421    }

423    /**
424     * Build the data archive of the deb from the provided DataProducers
425     *
426     * @param pData
427     * @param pOutput
428     * @param pChecksums
429     * @param pCompression the compression method used for the data file (gzip, bzip2 or anything else for no compression)
430     * @return
431     * @throws NoSuchAlgorithmException
432     * @throws IOException
433     * @throws CompressorException
434     */
435    BigInteger buildData( final DataProducer[] pData, final File pOutput, final StringBuilder pChecksums, String pCompression ) throws NoSuchAlgorithmException, IOException, CompressorException {
436
437        final File dir = pOutput.getParentFile();
438        if (dir != null && (!dir.exists() || !dir.isDirectory())) {
439            throw new IOException("Cannot write data file at '" + pOutput.getAbsolutePath() + "'");
440        }
441
442        final TarArchiveOutputStream tarOutputStream = new TarArchiveOutputStream(compressedOutputStream(pCompression, new FileOutputStream(pOutput)));
443        tarOutputStream.setLongFileMode(TarArchiveOutputStream.LONGFILE_GNU);
444
445        final MessageDigest digest = MessageDigest.getInstance("MD5");
446
447        final Total dataSize = new Total();
448
449        final List addedDirectories = new ArrayList();
450        final DataConsumer receiver = new DataConsumer() {
451            public void onEachDir( String dirname, String linkname, String user, int uid, String group, int gid, int mode, long size ) throws IOException {
452                dirname = fixPath(dirname);
453
454                createParentDirectories((new File(dirname)).getParent(), user, uid, group, gid);
455
456                // The directory passed in explicitly by the caller also gets the passed-in mode.  (Unlike
457                // the parent directories for now.  See related comments at "int mode =" in
458                // createParentDirectories, including about a possible bug.)
459                createDirectory(dirname, user, uid, group, gid, mode, 0);
460
461                console.info("dir: " + dirname);
462            }
463
464            public void onEachFile( InputStream inputStream, String filename, String linkname, String user, int uid, String group, int gid, int mode, long size ) throws IOException {
465                filename = fixPath(filename);
466
467                createParentDirectories((new File(filename)).getParent(), user, uid, group, gid);
468
469                final TarArchiveEntry entry = new TarArchiveEntry(filename, true);
470
471                entry.setUserName(user);
472                entry.setUserId(uid);
473                entry.setGroupName(group);
474                entry.setGroupId(gid);
475                entry.setMode(mode);
476                entry.setSize(size);
477
478                tarOutputStream.putArchiveEntry(entry);
479
480                dataSize.add(size);
481                digest.reset();
482
483                Utils.copy(inputStream, new DigestOutputStream(tarOutputStream, digest));
484
485                final String md5 = Utils.toHex(digest.digest());
486
487                tarOutputStream.closeArchiveEntry();
488
489                console.info(
490                    "file:" + entry.getName() +
491                        " size:" + entry.getSize() +
492                        " mode:" + entry.getMode() +
493                        " linkname:" + entry.getLinkName() +
494                        " username:" + entry.getUserName() +
495                        " userid:" + entry.getUserId() +
496                        " groupname:" + entry.getGroupName() +
497                        " groupid:" + entry.getGroupId() +
498                        " modtime:" + entry.getModTime() +
499                        " md5: " + md5
500                );
501
502                // append to file md5 list
503                pChecksums.append(md5).append(" ").append(entry.getName()).append('\n');
504            }
505
506            public void onEachLink(String path, String linkName, boolean symlink, String user, int uid, String group, int gid) throws IOException {
507                path = fixPath(path);
508
509                createParentDirectories((new File(path)).getParent(), user, uid, group, gid);
510
511                final TarArchiveEntry entry = new TarArchiveEntry(path, symlink ? TarArchiveEntry.LF_SYMLINK : TarArchiveEntry.LF_LINK);
512                entry.setLinkName(linkName);
513
514                tarOutputStream.putArchiveEntry(entry);
515                tarOutputStream.closeArchiveEntry();
516
517                console.info(
518                    "link:" + entry.getName() + " linkname:" + entry.getLinkName()
519                 );
520            }
521
522            private String fixPath( String path ) {
523                // If we're receiving directory names from Windows, then we'll convert to use slash
524                // This does eliminate the ability to use of a backslash in a directory name on *NIX,
525                // but in practice, this is a non-issue
526                if (path.indexOf('\\') > -1) {
527                    path = path.replace('\\', '/');
528                }
529                // ensure the path is like : ./foo/bar
530                if (path.startsWith("/")) {
531                    path = "." + path;
532                } else if (!path.startsWith("./")) {
533                    path = "./" + path;
534                }
535                return path;
536            }
537
538            private void createDirectory( String directory, String user, int uid, String group, int gid, int mode, long size ) throws IOException {
539                // All dirs should end with "/" when created, or the test DebAndTaskTestCase.testTarFileSet() thinks its a file
540                // and so thinks it has the wrong permission.
541                // This consistency also helps when checking if a directory already exists in addedDirectories.
542
543                if (!directory.endsWith("/")) {
544                    directory += "/";
545                }
546
547                if (!addedDirectories.contains(directory)) {
548                    TarArchiveEntry entry = new TarArchiveEntry(directory, true);
549                    entry.setUserName(user);
550                    entry.setUserId(uid);
551                    entry.setGroupName(group);
552                    entry.setGroupId(gid);
553                    entry.setMode(mode);
554                    entry.setSize(size);
555
556                    tarOutputStream.putArchiveEntry(entry);
557                    tarOutputStream.closeArchiveEntry();
558                    addedDirectories.add(directory); // so addedDirectories consistently have "/" for finding duplicates.
559                }
560            }
561
562            private void createParentDirectories( String dirname, String user, int uid, String group, int gid ) throws IOException {
563                // Debian packages must have parent directories created
564                // before sub-directories or files can be installed.
565                // For example, if an entry of ./usr/lib/foo/bar existed
566                // in a .deb package, but the ./usr/lib/foo directory didn't
567                // exist, the package installation would fail.  The .deb must
568                // then have an entry for ./usr/lib/foo and then ./usr/lib/foo/bar
569
570                if (dirname == null) {
571                    return;
572                }
573
574                // The loop below will create entries for all parent directories
575                // to ensure that .deb packages will install correctly.
576                String[] pathParts = dirname.split("\\/");
577                String parentDir = "./";
578                for (int i = 1; i < pathParts.length; i++) {
579                    parentDir += pathParts[i] + "/";
580                    // Make it so the dirs can be traversed by users.
581                    // We could instead try something more granular, like setting the directory
582                    // permission to 'rx' for each of the 3 user/group/other read permissions
583                    // found on the file being added (ie, only if "other" has read
584                    // permission on the main node, then add o+rx permission on all the containing
585                    // directories, same w/ user & group), and then also we'd have to
586                    // check the parentDirs collection of those already added to
587                    // see if those permissions need to be similarly updated.  (Note, it hasn't
588                    // been demonstrated, but there might be a bug if a user specifically
589                    // requests a directory with certain permissions,
590                    // that has already been auto-created because it was a parent, and if so, go set
591                    // the user-requested mode on that directory instead of this automatic one.)
592                    // But for now, keeping it simple by making every dir a+rx.   Examples are:
593                    // drw-r----- fs/fs   # what you get with setMode(mode)
594                    // drwxr-xr-x fs/fs   # Usable. Too loose?
595                    int mode = TarArchiveEntry.DEFAULT_DIR_MODE;
596
597                    createDirectory(parentDir, user, uid, group, gid, mode, 0);
598                }
599            }
600        };
601
602        try {
603            for (int i = 0; i < pData.length; i++) {
604                final DataProducer data = pData[i];
605                data.produce(receiver);
606            }
607        } finally {
608            tarOutputStream.close();
609        }
610
611        console.info("Total size: " + dataSize);
612
613        return dataSize.count;
614    }

616    private static void addEntry( final String pName, final String pContent, final TarArchiveOutputStream pOutput ) throws IOException {
617        final byte[] data = pContent.getBytes("UTF-8");
618
619        final TarArchiveEntry entry = new TarArchiveEntry("./" + pName, true);
620        entry.setSize(data.length);
621        entry.setNames("root", "root");
622
623        pOutput.putArchiveEntry(entry);
624        pOutput.write(data);
625        pOutput.closeArchiveEntry();
626    }

628    private static void addControlEntry( final String pName, final String pContent, final TarArchiveOutputStream pOutput ) throws IOException {
629        final byte[] data = pContent.getBytes("UTF-8");
630
631        final TarArchiveEntry entry = new TarArchiveEntry("./" + pName, true);
632        entry.setSize(data.length);
633        entry.setNames("root", "root");
634        entry.setMode(PermMapper.toMode("755"));
635
636        pOutput.putArchiveEntry(entry);
637        pOutput.write(data);
638        pOutput.closeArchiveEntry();
639    }
