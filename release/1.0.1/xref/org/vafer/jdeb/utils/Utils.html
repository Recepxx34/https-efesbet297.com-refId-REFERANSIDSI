38      public static int copy( final InputStream pInput, final OutputStream pOutput ) throws IOException {
39          final byte[] buffer = new byte[2048];
40          int count = 0;
41          int n;
42          while (-1 != (n = pInput.read(buffer))) {
43              pOutput.write(buffer, 0, n);
44              count += n;
45          }
46          return count;
47      }

49      public static String toHex( final byte[] bytes ) {
50          final StringBuilder sb = new StringBuilder();
51  
52          for (byte b : bytes) {
53              sb.append(Integer.toHexString((b >> 4) & 0x0f));
54              sb.append(Integer.toHexString(b & 0x0f));
55          }
56  
57          return sb.toString();
58      }

60      public static String stripPath( final int p, final String s ) {
61  
62          if (p <= 0) {
63              return s;
64          }
65  
66          int x = 0;
67          for (int i = 0; i < p; i++) {
68              x = s.indexOf('/', x + 1);
69              if (x < 0) {
70                  return s;
71              }
72          }
73  
74          return s.substring(x + 1);
75      }
76  
77      public static String stripLeadingSlash( final String s ) {
78          if (s == null) {
79              return s;
80          }
81          if (s.length() == 0) {
82              return s;
83          }
84          if (s.charAt(0) == '/' || s.charAt(0) == '\\') {
85              return s.substring(1);
86          }
87          return s;
88      }

91      /**
92       * Substitute the variables in the given expression with the
93       * values from the resolver
94       *
95       * @param pResolver
96       * @param pExpression
97       */
98      public static String replaceVariables( final VariableResolver pResolver, final String pExpression, final String pOpen, final String pClose ) {
99          final char[] open = pOpen.toCharArray();
100         final char[] close = pClose.toCharArray();
101         
102         final StringBuilder out = new StringBuilder();
103         StringBuilder sb = new StringBuilder();
104         char[] watch = open;
105         int w = 0;
106         for (char c : pExpression.toCharArray()) {
107             if (c == watch[w]) {
108                 w++;
109                 if (watch.length == w) {
110                     // found the full token to watch for
111 
112                     if (watch == open) {
113                         // found open
114                         out.append(sb);
115                         sb = new StringBuilder();
116                         // search for close
117                         watch = close;
118                     } else {
119                         // found close
120                         final String variable = pResolver.get(sb.toString());
121                         if (variable != null) {
122                             out.append(variable);
123                         } else {
124                             out.append(pOpen);
125                             out.append(sb);
126                             out.append(pClose);
127                         }
128                         sb = new StringBuilder();
129                         // search for open
130                         watch = open;
131                     }
132                     w = 0;
133                 }
134             } else {
135 
136                 if (w > 0) {
137                     sb.append(watch, 0, w);
138                 }
139 
140                 sb.append(c);
141 
142                 w = 0;
143             }
144         }
145 
146         if (watch == close) {
147             out.append(open);
148         }
149         out.append(sb);
150 
151         return out.toString();
152     }

154     /**
155      * Replaces new line delimiters in the input stream with the Unix line feed.
156      *
157      * @param input
158      */
159     public static byte[] toUnixLineEndings( InputStream input ) throws IOException {
160         String encoding = "ISO-8859-1";
161         FixCrLfFilter filter = new FixCrLfFilter(new InputStreamReader(input, encoding));
162         filter.setEol(FixCrLfFilter.CrLf.newInstance("unix"));
163 
164         ByteArrayOutputStream filteredFile = new ByteArrayOutputStream();
165         Utils.copy(new ReaderInputStream(filter, encoding), filteredFile);
166 
167         return filteredFile.toByteArray();
168     }

170     /**
171      * convert to debian version format
172      */
173     public static String convertToDebianVersion( String version, Date timestamp ) {
174         version = version.replace('-', '+');
175         if (version.endsWith("+SNAPSHOT")) {
176             version = version.substring(0, version.length() - "+SNAPSHOT".length());
177             version += "~";
178 
179             if (timestamp != null) {
180                 version += new SimpleDateFormat("yyyyMMddHHmmss").format(timestamp);
181             } else {
182                 version += "SNAPSHOT";
183             }
184         }
185         return version;
186     }
