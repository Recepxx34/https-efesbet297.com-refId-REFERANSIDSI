016    package org.vafer.jdeb.producers;
017    
018    import java.io.BufferedInputStream;
019    import java.io.File;
020    import java.io.FileInputStream;
021    import java.io.IOException;
022    import java.io.InputStream;
023    
024    import org.apache.commons.compress.archivers.ArchiveEntry;
025    import org.apache.commons.compress.archivers.ArchiveException;
026    import org.apache.commons.compress.archivers.ArchiveInputStream;
027    import org.apache.commons.compress.archivers.ArchiveStreamFactory;
028    import org.apache.commons.compress.archivers.tar.TarArchiveEntry;
029    import org.apache.commons.compress.archivers.tar.TarArchiveInputStream;
030    import org.apache.commons.compress.archivers.zip.ZipArchiveEntry;
031    import org.apache.commons.compress.archivers.zip.ZipArchiveInputStream;
032    import org.apache.commons.compress.compressors.CompressorException;
033    import org.apache.commons.compress.compressors.CompressorInputStream;
034    import org.apache.commons.compress.compressors.CompressorStreamFactory;
035    import org.apache.tools.tar.TarEntry;
036    import org.vafer.jdeb.DataConsumer;
037    import org.vafer.jdeb.DataProducer;
038    import org.vafer.jdeb.mapping.Mapper;
039    
040    /**
041     * Providing data from an archive keeping permissions and ownerships.
042     *
043     * @author Torsten Curdt <tcurdt@vafer.org>
044     */
045    public final class DataProducerArchive extends AbstractDataProducer implements DataProducer {
046    
047        private final File archive;
048    
049        public DataProducerArchive( final File pArchive, final String[] pIncludes, final String[] pExcludes, final Mapper[] pMappers ) {
050            super(pIncludes, pExcludes, pMappers);
051            archive = pArchive;
052        }
053    
054        public void produce( final DataConsumer pReceiver ) throws IOException {
055    
056            InputStream is = new BufferedInputStream(new FileInputStream(archive));
057    
058            CompressorInputStream compressorInputStream = null;
059    
060            try {
061                compressorInputStream = new CompressorStreamFactory().createCompressorInputStream(is);
062            } catch(CompressorException e) {
063                // expected if the input file is a zip archive
064            }
065    
066            if (compressorInputStream != null) {
067                is = new BufferedInputStream(compressorInputStream);
068            }
069    
070            ArchiveInputStream archiveInputStream = null;
071    
072            try {
073                archiveInputStream = new ArchiveStreamFactory().createArchiveInputStream(is);
074            } catch(ArchiveException e) {
075                throw new IOException("Unsupported archive format: " + archive, e);
076            }
077    
078            EntryConverter converter = null;
079    
080            if (archiveInputStream instanceof TarArchiveInputStream) {
081    
082                converter = new EntryConverter() {
083                    public TarEntry convert(ArchiveEntry entry) {
084                        TarArchiveEntry src = (TarArchiveEntry)entry;
085                        TarEntry dst = new TarEntry(src.getName());
086    
087                        dst.setSize(src.getSize());
088                        dst.setGroupName(src.getGroupName());
089                        dst.setGroupId(src.getGroupId());
090                        dst.setUserId(src.getUserId());
091                        dst.setMode(src.getMode());
092                        dst.setModTime(src.getModTime());
093    
094                        return dst;
095                    }
096                };
097    
098            } else if (archiveInputStream instanceof ZipArchiveInputStream) {
099    
100                converter = new EntryConverter() {
101                    public TarEntry convert(ArchiveEntry entry) {
102                        ZipArchiveEntry src = (ZipArchiveEntry)entry;
103                        TarEntry dst = new TarEntry(src.getName());
104    
105                        dst.setSize(src.getSize());
106                        dst.setMode(src.getUnixMode());
107                        dst.setModTime(src.getTime());
108    
109                        return dst;
110                    }
111                };
112    
113            } else {
114                throw new IOException("Unsupported archive format: " + archive);
115            }
116    
117    
118            try {
119                while(true) {
120    
121                    ArchiveEntry archiveEntry = archiveInputStream.getNextEntry();
122    
123                    if (archiveEntry == null) {
124                        break;
125                    }
126    
127                    if (!isIncluded(archiveEntry.getName())) {
128                        continue;
129                    }
130    
131                    TarEntry entry = converter.convert(archiveEntry);
132    
133                    entry = map(entry);
134    
135                    if (entry.isDirectory()) {
136                        pReceiver.onEachDir(entry.getName(), entry.getLinkName(), entry.getUserName(), entry.getUserId(), entry.getGroupName(), entry.getGroupId(), entry.getMode(), entry.getSize());
137                        continue;
138                    }
139                    pReceiver.onEachFile(archiveInputStream, entry.getName(), entry.getLinkName(), entry.getUserName(), entry.getUserId(), entry.getGroupName(), entry.getGroupId(), entry.getMode(), entry.getSize());
140                }
141    
142            } finally {
143                if (archiveInputStream != null) {
144                    archiveInputStream.close();
145                }
146            }
147        }
148    
149        private interface EntryConverter {
150            public TarEntry convert(ArchiveEntry entry);
151        }
152    }
