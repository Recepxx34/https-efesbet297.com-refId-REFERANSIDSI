095    private void addTo( final ArArchiveOutputStream pOutput, final String pName, final String pContent ) throws IOException {
096        final byte[] content = pContent.getBytes();
097        pOutput.putArchiveEntry(new ArArchiveEntry(pName, content.length));
098        pOutput.write(content);
099        pOutput.closeArchiveEntry();
100    }
101
102    private void addTo( final ArArchiveOutputStream pOutput, final String pName, final File pContent ) throws IOException {
103        pOutput.putArchiveEntry(new ArArchiveEntry(pName, pContent.length()));
104
105        final InputStream input = new FileInputStream(pContent);
106        try {
107            Utils.copy(input, pOutput);
108        } finally {
109            input.close();
110        }
111
112        pOutput.closeArchiveEntry();
113    }

115    /**
116     * Create the debian archive with from the provided control files and data producers.
117     *
118     * @param pControlFiles
119     * @param pData
120     * @param pOutput
121     * @param compression   the compression method used for the data file
122     * @return PackageDescriptor
123     * @throws PackagingException
124     */
125    public PackageDescriptor createDeb( final File[] pControlFiles, final DataProducer[] pData, final File pOutput, Compression compression ) throws PackagingException {
126
127        File tempData = null;
128        File tempControl = null;
129
130        try {
131            tempData = File.createTempFile("deb", "data");
132            tempControl = File.createTempFile("deb", "control");
133
134            console.info("Building data");
135            final StringBuilder md5s = new StringBuilder();
136            final BigInteger size = buildData(pData, tempData, md5s, compression);
137
138            console.info("Building control");
139            final PackageDescriptor packageDescriptor = buildControl(pControlFiles, size, md5s, tempControl);
140
141            if (!packageDescriptor.isValid()) {
142                throw new PackagingException("Control file descriptor keys are invalid " + packageDescriptor.invalidKeys() +
143                    ". The following keys are mandatory " + Arrays.toString(PackageDescriptor.MANDATORY_KEYS) +
144                    ". Please check your pom.xml/build.xml and your control file.");
145            }
146
147            pOutput.getParentFile().mkdirs();
148
149            // pass through stream chain to calculate all the different digests
150            final InformationOutputStream md5output = new InformationOutputStream(new FileOutputStream(pOutput), MessageDigest.getInstance("MD5"));
151            final InformationOutputStream sha1output = new InformationOutputStream(md5output, MessageDigest.getInstance("SHA1"));
152            final InformationOutputStream sha256output = new InformationOutputStream(sha1output, MessageDigest.getInstance("SHA-256"));
153            final ArArchiveOutputStream ar = new ArArchiveOutputStream(sha256output);
154
155            addTo(ar, "debian-binary", "2.0\n");
156            addTo(ar, "control.tar.gz", tempControl);
157            addTo(ar, "data.tar" + compression.getExtension(), tempData);
158
159            ar.close();
160
161            // intermediate values
162            packageDescriptor.set("MD5", md5output.getHexDigest());
163            packageDescriptor.set("SHA1", sha1output.getHexDigest());
164            packageDescriptor.set("SHA256", sha256output.getHexDigest());
165            packageDescriptor.set("Size", "" + md5output.getSize());
166            packageDescriptor.set("File", pOutput.getName());
167
168            return packageDescriptor;
169
170        } catch (Exception e) {
171            throw new PackagingException("Could not create deb package", e);
172        } finally {
173            if (tempData != null) {
174                if (!tempData.delete()) {
175                    console.warn("Could not delete the temporary file " + tempData);
176                }
177            }
178            if (tempControl != null) {
179                if (!tempControl.delete()) {
180                    console.warn("Could not delete the temporary file " + tempControl);
181                }
182            }
183        }
184    }

186    /**
187     * Create changes file based on the provided PackageDescriptor.
188     * If pRing, pKey and pPassphrase are provided the changes file will also be signed.
189     * It returns a ChangesDescriptor reflecting the changes
190     *
191     * @param pPackageDescriptor
192     * @param pChangesProvider
193     * @param pRing
194     * @param pKey
195     * @param pPassphrase
196     * @param pOutput
197     * @return ChangesDescriptor
198     * @throws IOException
199     * @throws PackagingException
200     */
201    public ChangesDescriptor createChanges( final PackageDescriptor pPackageDescriptor, final ChangesProvider pChangesProvider, final InputStream pRing, final String pKey, final String pPassphrase, final OutputStream pOutput ) throws IOException, PackagingException {
202
203        final ChangeSet[] changeSets = pChangesProvider.getChangesSets();
204        final ChangesDescriptor changesDescriptor = new ChangesDescriptor(pPackageDescriptor, changeSets);
205
206        changesDescriptor.set("Format", "1.8");
207
208        if (changesDescriptor.get("Binary") == null) {
209            changesDescriptor.set("Binary", changesDescriptor.get("Package"));
210        }
211
212        if (changesDescriptor.get("Source") == null) {
213            changesDescriptor.set("Source", changesDescriptor.get("Package"));
214        }
215
216        final StringBuilder checksumsSha1 = new StringBuilder("\n");
217        // Checksums-Sha1:
218        // 56ef4c6249dc3567fd2967f809c42d1f9b61adf7 45964 jdeb.deb
219        checksumsSha1.append(' ').append(changesDescriptor.get("SHA1"));
220        checksumsSha1.append(' ').append(changesDescriptor.get("Size"));
221        checksumsSha1.append(' ').append(changesDescriptor.get("File"));
222        changesDescriptor.set("Checksums-Sha1", checksumsSha1.toString());
223
224        final StringBuilder checksumsSha256 = new StringBuilder("\n");
225        // Checksums-Sha256:
226        // 38c6fa274eb9299a69b739bcbdbd05c7ffd1d8d6472f4245ed732a25c0e5d616 45964 jdeb.deb
227        checksumsSha256.append(' ').append(changesDescriptor.get("SHA256"));
228        checksumsSha256.append(' ').append(changesDescriptor.get("Size"));
229        checksumsSha256.append(' ').append(changesDescriptor.get("File"));
230        changesDescriptor.set("Checksums-Sha256", checksumsSha256.toString());
231
232
233        final StringBuilder files = new StringBuilder("\n");
234        files.append(' ').append(changesDescriptor.get("MD5"));
235        files.append(' ').append(changesDescriptor.get("Size"));
236        files.append(' ').append(changesDescriptor.get("Section"));
237        files.append(' ').append(changesDescriptor.get("Priority"));
238        files.append(' ').append(changesDescriptor.get("File"));
239        changesDescriptor.set("Files", files.toString());
240
241        if (!changesDescriptor.isValid()) {
242            throw new PackagingException("Changes file descriptor keys are invalid " + changesDescriptor.invalidKeys() +
243                ". The following keys are mandatory " + Arrays.toString(ChangesDescriptor.MANDATORY_KEYS) +
244                ". Please check your pom.xml/build.xml and your control file.");
245        }
246
247        final String changes = changesDescriptor.toString();
248        final byte[] changesBytes = changes.getBytes("UTF-8");
249
250        if (pRing == null || pKey == null || pPassphrase == null) {
251            pOutput.write(changesBytes);
252            pOutput.close();
253            return changesDescriptor;
254        }
255
256        console.info("Signing changes with key " + pKey);
257
258        final InputStream input = new ByteArrayInputStream(changesBytes);
259
260        try {
261            SigningUtils.clearSign(input, pRing, pKey, pPassphrase, pOutput);
262        } catch (Exception e) {
263            e.printStackTrace();
264        }
265
266        pOutput.close();
267
268        return changesDescriptor;
269    }

271    /**
272     * Build control archive of the deb
273     *
274     * @param pControlFiles
275     * @param pDataSize
276     * @param pChecksums
277     * @param pOutput
278     * @return
279     * @throws FileNotFoundException
280     * @throws IOException
281     * @throws ParseException
282     */
283    private PackageDescriptor buildControl( final File[] pControlFiles, final BigInteger pDataSize, final StringBuilder pChecksums, final File pOutput ) throws IOException, ParseException {
284
285        final File dir = pOutput.getParentFile();
286        if (dir != null && (!dir.exists() || !dir.isDirectory())) {
287            throw new IOException("Cannot write control file at '" + pOutput.getAbsolutePath() + "'");
288        }
289
290        final TarArchiveOutputStream outputStream = new TarArchiveOutputStream(new GZIPOutputStream(new FileOutputStream(pOutput)));
291        outputStream.setLongFileMode(TarArchiveOutputStream.LONGFILE_GNU);
292        
293        List<FilteredConfigurationFile> configurationFiles = new ArrayList<FilteredConfigurationFile>();
294        
295        // create a descriptor out of the "control" file, copy all other files, ignore directories
296        PackageDescriptor packageDescriptor = null;
297        for (File file : pControlFiles) {
298            if (file.isDirectory()) {
299                // warn about the misplaced directory, except for directories ignored by default (.svn, cvs, etc)
300                if (!isDefaultExcludes(file)) {
301                    console.info("Found directory '" + file + "' in the control directory. Maybe you are pointing to wrong dir?");
302                }
303                continue;
304            }
305
306            if (CONFIGURATION_FILENAMES.contains(file.getName()) || MAINTAINER_SCRIPTS.contains(file.getName())) {
307                FilteredConfigurationFile configurationFile = new FilteredConfigurationFile(file.getName(), new FileInputStream(file), resolver);
308                configurationFiles.add(configurationFile);
309
310            } else if ("control".equals(file.getName())) {
311                packageDescriptor = createPackageDescriptor(file, pDataSize);
312
313            } else {
314                // initialize the information stream to guess the type of the file
315                InformationInputStream infoStream = new InformationInputStream(new FileInputStream(file));
316                Utils.copy(infoStream, NullOutputStream.NULL_OUTPUT_STREAM);
317                infoStream.close();
318
319                // fix line endings for shell scripts
320                InputStream in = new FileInputStream(file);
321                if (infoStream.isShell() && !infoStream.hasUnixLineEndings()) {
322                    byte[] buf = Utils.toUnixLineEndings(in);
323                    in = new ByteArrayInputStream(buf);
324                }
325                
326                addControlEntry(file.getName(), IOUtils.toString(in), outputStream);
327                
328                in.close();
329            }
330        }
331
332        if (packageDescriptor == null) {
333            throw new FileNotFoundException("No 'control' file found in " + Arrays.toString(pControlFiles));
334        }
335
336        for (FilteredConfigurationFile configurationFile : configurationFiles) {
337            addControlEntry(configurationFile.getName(), configurationFile.toString(), outputStream);
338        }
339        addControlEntry("control", packageDescriptor.toString(), outputStream);
340        addControlEntry("md5sums", pChecksums.toString(), outputStream);
341
342        outputStream.close();
343
344        return packageDescriptor;
345    }

347    /**
348     * Tells if the specified directory is ignored by default (.svn, cvs, etc)
349     * 
350     * @param directory
351     */
352    private boolean isDefaultExcludes(File directory) {
353        for (String pattern : DirectoryScanner.getDefaultExcludes()) {
354            if (DirectoryScanner.match(pattern, directory.getAbsolutePath().replace("\\", "/"))) {
355                return true;
356            }
357        }
358        
359        return false;
360    }

362    /**
363     * Creates a package descriptor from the specified control file and adds the
364     * <tt>Date</tt>, <tt>Distribution</tt> and <tt>Urgency</tt> fields if missing.
365     * The <tt>Installed-Size</tt> field is also initialized to the actual size of
366     * the package. The <tt>Maintainer</tt> field is overridden by the <tt>DEBEMAIL</tt>
367     * and <tt>DEBFULLNAME</tt> environment variables if defined.
368     * 
369     * @param file       the control file
370     * @param pDataSize  the size of the installed package
371     */
372    private PackageDescriptor createPackageDescriptor(File file, BigInteger pDataSize) throws IOException, ParseException {
373        PackageDescriptor packageDescriptor = new PackageDescriptor(new FileInputStream(file), resolver);
374
375        if (packageDescriptor.get("Date") == null) {
376            // Mon, 26 Mar 2007 11:44:04 +0200 (RFC 2822)
377            SimpleDateFormat fmt = new SimpleDateFormat("EEE, d MMM yyyy HH:mm:ss Z", Locale.ENGLISH);
378            packageDescriptor.set("Date", fmt.format(new Date()));
379        }
380
381        if (packageDescriptor.get("Distribution") == null) {
382            packageDescriptor.set("Distribution", "unknown");
383        }
384
385        if (packageDescriptor.get("Urgency") == null) {
386            packageDescriptor.set("Urgency", "low");
387        }
388
389        packageDescriptor.set("Installed-Size", pDataSize.divide(BigInteger.valueOf(1024)).toString());
390
391        // override the Version if the DEBVERSION environment variable is defined
392        final String debVersion = System.getenv("DEBVERSION");
393        if (debVersion != null) {
394            packageDescriptor.set("Version", debVersion);
395            console.info("Using version'" + debVersion + "' from the environment variables.");
396        }
397
398
399        // override the Maintainer field if the DEBFULLNAME and DEBEMAIL environment variables are defined
400        final String debFullName = System.getenv("DEBFULLNAME");
401        final String debEmail = System.getenv("DEBEMAIL");
402
403        if (debFullName != null && debEmail != null) {
404            final String maintainer = debFullName + " <" + debEmail + ">";
405            packageDescriptor.set("Maintainer", maintainer);
406            console.info("Using maintainer '" + maintainer + "' from the environment variables.");
407        }
408        
409        return packageDescriptor;
410    }

412    /**
413     * Build the data archive of the deb from the provided DataProducers
414     *
415     * @param pData
416     * @param pOutput
417     * @param pChecksums
418     * @param compression the compression method used for the data file
419     * @return
420     * @throws NoSuchAlgorithmException
421     * @throws IOException
422     * @throws CompressorException
423     */
424    BigInteger buildData( final DataProducer[] pData, final File pOutput, final StringBuilder pChecksums, Compression compression ) throws NoSuchAlgorithmException, IOException, CompressorException {
425
426        final File dir = pOutput.getParentFile();
427        if (dir != null && (!dir.exists() || !dir.isDirectory())) {
428            throw new IOException("Cannot write data file at '" + pOutput.getAbsolutePath() + "'");
429        }
430
431        final TarArchiveOutputStream tarOutputStream = new TarArchiveOutputStream(compression.toCompressedOutputStream(new FileOutputStream(pOutput)));
432        tarOutputStream.setLongFileMode(TarArchiveOutputStream.LONGFILE_GNU);
433
434        final MessageDigest digest = MessageDigest.getInstance("MD5");
435
436        final Total dataSize = new Total();
437
438        final List<String> addedDirectories = new ArrayList<String>();
439        final DataConsumer receiver = new DataConsumer() {
440            public void onEachDir( String dirname, String linkname, String user, int uid, String group, int gid, int mode, long size ) throws IOException {
441                dirname = fixPath(dirname);
442
443                createParentDirectories((new File(dirname)).getParent(), user, uid, group, gid);
444
445                // The directory passed in explicitly by the caller also gets the passed-in mode.  (Unlike
446                // the parent directories for now.  See related comments at "int mode =" in
447                // createParentDirectories, including about a possible bug.)
448                createDirectory(dirname, user, uid, group, gid, mode, 0);
449
450                console.info("dir: " + dirname);
451            }
452
453            public void onEachFile( InputStream inputStream, String filename, String linkname, String user, int uid, String group, int gid, int mode, long size ) throws IOException {
454                filename = fixPath(filename);
455
456                createParentDirectories((new File(filename)).getParent(), user, uid, group, gid);
457
458                final TarArchiveEntry entry = new TarArchiveEntry(filename, true);
459
460                entry.setUserName(user);
461                entry.setUserId(uid);
462                entry.setGroupName(group);
463                entry.setGroupId(gid);
464                entry.setMode(mode);
465                entry.setSize(size);
466
467                tarOutputStream.putArchiveEntry(entry);
468
469                dataSize.add(size);
470                digest.reset();
471
472                Utils.copy(inputStream, new DigestOutputStream(tarOutputStream, digest));
473
474                final String md5 = Utils.toHex(digest.digest());
475
476                tarOutputStream.closeArchiveEntry();
477
478                console.info(
479                    "file:" + entry.getName() +
480                        " size:" + entry.getSize() +
481                        " mode:" + entry.getMode() +
482                        " linkname:" + entry.getLinkName() +
483                        " username:" + entry.getUserName() +
484                        " userid:" + entry.getUserId() +
485                        " groupname:" + entry.getGroupName() +
486                        " groupid:" + entry.getGroupId() +
487                        " modtime:" + entry.getModTime() +
488                        " md5: " + md5
489                );
490
491                // append to file md5 list
492                pChecksums.append(md5).append(" ").append(entry.getName()).append('\n');
493            }
494
495            public void onEachLink(String path, String linkName, boolean symlink, String user, int uid, String group, int gid) throws IOException {
496                path = fixPath(path);
497
498                createParentDirectories((new File(path)).getParent(), user, uid, group, gid);
499
500                final TarArchiveEntry entry = new TarArchiveEntry(path, symlink ? TarArchiveEntry.LF_SYMLINK : TarArchiveEntry.LF_LINK);
501                entry.setLinkName(linkName);
502
503                tarOutputStream.putArchiveEntry(entry);
504                tarOutputStream.closeArchiveEntry();
505
506                console.info(
507                    "link:" + entry.getName() + " linkname:" + entry.getLinkName()
508                 );
509            }
510
511            private String fixPath( String path ) {
512                // If we're receiving directory names from Windows, then we'll convert to use slash
513                // This does eliminate the ability to use of a backslash in a directory name on *NIX,
514                // but in practice, this is a non-issue
515                if (path.indexOf('\\') > -1) {
516                    path = path.replace('\\', '/');
517                }
518                // ensure the path is like : ./foo/bar
519                if (path.startsWith("/")) {
520                    path = "." + path;
521                } else if (!path.startsWith("./")) {
522                    path = "./" + path;
523                }
524                return path;
525            }
526
527            private void createDirectory( String directory, String user, int uid, String group, int gid, int mode, long size ) throws IOException {
528                // All dirs should end with "/" when created, or the test DebAndTaskTestCase.testTarFileSet() thinks its a file
529                // and so thinks it has the wrong permission.
530                // This consistency also helps when checking if a directory already exists in addedDirectories.
531
532                if (!directory.endsWith("/")) {
533                    directory += "/";
534                }
535
536                if (!addedDirectories.contains(directory)) {
537                    TarArchiveEntry entry = new TarArchiveEntry(directory, true);
538                    entry.setUserName(user);
539                    entry.setUserId(uid);
540                    entry.setGroupName(group);
541                    entry.setGroupId(gid);
542                    entry.setMode(mode);
543                    entry.setSize(size);
544
545                    tarOutputStream.putArchiveEntry(entry);
546                    tarOutputStream.closeArchiveEntry();
547                    addedDirectories.add(directory); // so addedDirectories consistently have "/" for finding duplicates.
548                }
549            }
550
551            private void createParentDirectories( String dirname, String user, int uid, String group, int gid ) throws IOException {
552                // Debian packages must have parent directories created
553                // before sub-directories or files can be installed.
554                // For example, if an entry of ./usr/lib/foo/bar existed
555                // in a .deb package, but the ./usr/lib/foo directory didn't
556                // exist, the package installation would fail.  The .deb must
557                // then have an entry for ./usr/lib/foo and then ./usr/lib/foo/bar
558
559                if (dirname == null) {
560                    return;
561                }
562
563                // The loop below will create entries for all parent directories
564                // to ensure that .deb packages will install correctly.
565                String[] pathParts = dirname.split("\\/");
566                String parentDir = "./";
567                for (int i = 1; i < pathParts.length; i++) {
568                    parentDir += pathParts[i] + "/";
569                    // Make it so the dirs can be traversed by users.
570                    // We could instead try something more granular, like setting the directory
571                    // permission to 'rx' for each of the 3 user/group/other read permissions
572                    // found on the file being added (ie, only if "other" has read
573                    // permission on the main node, then add o+rx permission on all the containing
574                    // directories, same w/ user & group), and then also we'd have to
575                    // check the parentDirs collection of those already added to
576                    // see if those permissions need to be similarly updated.  (Note, it hasn't
577                    // been demonstrated, but there might be a bug if a user specifically
578                    // requests a directory with certain permissions,
579                    // that has already been auto-created because it was a parent, and if so, go set
580                    // the user-requested mode on that directory instead of this automatic one.)
581                    // But for now, keeping it simple by making every dir a+rx.   Examples are:
582                    // drw-r----- fs/fs   # what you get with setMode(mode)
583                    // drwxr-xr-x fs/fs   # Usable. Too loose?
584                    int mode = TarArchiveEntry.DEFAULT_DIR_MODE;
585
586                    createDirectory(parentDir, user, uid, group, gid, mode, 0);
587                }
588            }
589        };
590
591        try {
592            for (DataProducer data : pData) {
593                data.produce(receiver);
594            }
595        } finally {
596            tarOutputStream.close();
597        }
598
599        console.info("Total size: " + dataSize);
600
601        return dataSize.count;
602    }

604    private static void addControlEntry(final String pName, final String pContent, final TarArchiveOutputStream pOutput) throws IOException {
605        final byte[] data = pContent.getBytes("UTF-8");
606
607        final TarArchiveEntry entry = new TarArchiveEntry("./" + pName, true);
608        entry.setSize(data.length);
609        entry.setNames("root", "root");
610        
611        if (MAINTAINER_SCRIPTS.contains(pName)) {
612            entry.setMode(PermMapper.toMode("755"));
613        } else {
614            entry.setMode(PermMapper.toMode("644"));
615        }
616        
617        pOutput.putArchiveEntry(entry);
618        pOutput.write(data);
619        pOutput.closeArchiveEntry();
620    }
