018import java.io.BufferedReader;
019import java.io.IOException;

021import java.io.InputStreamReader;

029/**
030 * Reads permissions and ownerships from a "ls -laR > mapping.txt" dump and
031 * maps entries accordingly.
032 *
033 * @author Torsten Curdt
034 */
035public final class LsMapper implements Mapper {
036
037    private final Map mapping;
038
039
040    public final static class ParseError extends Exception {
041
042        private static final long serialVersionUID = 1L;
043
044        public ParseError() {
045            super();
046        }
047
048        public ParseError( String message, Throwable cause ) {
049            super(message, cause);
050        }
051
052        public ParseError( String message ) {
053            super(message);
054        }
055
056        public ParseError( Throwable cause ) {
057            super(cause);
058        }
059
060    }
061
062    ;
063
064    public LsMapper( final InputStream pInput ) throws IOException, ParseError {
065        mapping = parse(pInput);
066    }

081    final private Pattern basePattern = Pattern.compile("^\\./(.*):$");
082    final private Pattern totalPattern = Pattern.compile("^total ([0-9]+)$");
083    final private Pattern dirPattern = Pattern.compile("^d([rwx-]{9})\\s+([0-9]+)\\s+(\\S*)\\s+(\\S*)\\s+([0-9]+)\\s+(.*)\\s+[\\.]{1,2}$");
084    final private Pattern filePattern = Pattern.compile("^([d-])([rwx-]{9})\\s+([0-9]+)\\s+(\\S*)\\s+(\\S*)\\s+([0-9]+)\\s+(.*)\\s+(.*)$");
085    final private Pattern newlinePattern = Pattern.compile("$");
086
087    private String readBase( final BufferedReader reader ) throws IOException, ParseError {
088        final String line = reader.readLine();
089        if (line == null) {
090            return null;
091        }
092        final Matcher matcher = basePattern.matcher(line);
093        if (!matcher.matches()) {
094            throw new ParseError("expected base line but got \"" + line + "\"");
095        }
096        return matcher.group(1);
097    }
098
099    private String readTotal( final BufferedReader reader ) throws IOException, ParseError {
100        final String line = reader.readLine();
101        final Matcher matcher = totalPattern.matcher(line);
102        if (!matcher.matches()) {
103            throw new ParseError("expected total line but got \"" + line + "\"");
104        }
105        return matcher.group(1);
106    }
107
108    private TarArchiveEntry readDir( final BufferedReader reader, final String base ) throws IOException, ParseError {
109        final String current = reader.readLine();
110        final Matcher currentMatcher = dirPattern.matcher(current);
111        if (!currentMatcher.matches()) {
112            throw new ParseError("expected dirline but got \"" + current + "\"");
113        }
114
115        final String parent = reader.readLine();
116        final Matcher parentMatcher = dirPattern.matcher(parent);
117        if (!parentMatcher.matches()) {
118            throw new ParseError("expected dirline but got \"" + parent + "\"");
119        }
120
121        final TarArchiveEntry entry = new TarArchiveEntry(base, true);
122
123        entry.setMode(convertModeFromString(currentMatcher.group(1)));
124        entry.setUserName(currentMatcher.group(3));
125        entry.setGroupName(currentMatcher.group(4));
126
127        return entry;
128    }
129
130
131    private int convertModeFromString( final String mode ) {
132
133        final char[] m = mode.toCharArray();
134        /*
135           -rwxrwxrwx
136
137           4000    set-user-ID-on-execution bit
138           2000    set-user-ID-on-execution bit
139           1000    sticky bit
140           0400    allow read by owner.
141           0200    allow write by owner.
142           0100    execute / search
143           0040    allow read by group members.
144           0020    allow write by group members.
145           0010    execute / search
146           0004    allow read by others.
147           0002    allow write by others.
148           0001    execute / search
149         */
150        // TODO: simplified - needs fixing
151        int sum = 0;
152        int bit = 1;
153        for (int i = m.length - 1; i >= 0; i--) {
154            if (m[i] != '-') {
155                sum += bit;
156            }
157            bit += bit;
158        }
159        return sum;
160    }
161
162    private TarArchiveEntry readFile( final BufferedReader reader, final String base ) throws IOException, ParseError {
163
164        while (true) {
165            final String line = reader.readLine();
166
167            if (line == null) {
168                return null;
169            }
170
171            final Matcher currentMatcher = filePattern.matcher(line);
172            if (!currentMatcher.matches()) {
173                final Matcher newlineMatcher = newlinePattern.matcher(line);
174                if (newlineMatcher.matches()) {
175                    return null;
176                }
177                throw new ParseError("expected file line but got \"" + line + "\"");
178            }
179
180            final String type = currentMatcher.group(1);
181            if (type.startsWith("-")) {
182                final TarArchiveEntry entry = new TarArchiveEntry(base + "/" + currentMatcher.group(8), true);
183
184                entry.setMode(convertModeFromString(currentMatcher.group(2)));
185                entry.setUserName(currentMatcher.group(4));
186                entry.setGroupName(currentMatcher.group(5));
187
188                return entry;
189            }
190        }
191
192    }
193
194    private Map parse( final InputStream pInput ) throws IOException, ParseError {
195        final Map mapping = new HashMap();
196
197        final BufferedReader reader = new BufferedReader(new InputStreamReader(pInput));
198
199        boolean first = true;
200        while (true) {
201
202            final String base;
203            if (first) {
204                base = "";
205                first = false;
206            } else {
207                base = readBase(reader);
208                if (base == null) {
209                    break;
210                }
211            }
212
213            readTotal(reader);
214            final TarArchiveEntry dir = readDir(reader, base);
215            mapping.put(dir.getName(), dir);
216
217            while (true) {
218                final TarArchiveEntry file = readFile(reader, base);
219
220                if (file == null) {
221                    break;
222                }
223
224                mapping.put(file.getName(), file);
225            }
226        }
227
228        return mapping;
229    }
230
231    public TarArchiveEntry map( final TarArchiveEntry pEntry ) {
232
233        final TarArchiveEntry entry = (TarArchiveEntry) mapping.get(pEntry.getName());
234
235        if (entry != null) {
236
237            final TarArchiveEntry newEntry = new TarArchiveEntry(entry.getName(), true);
238            newEntry.setUserId(entry.getUserId());
239            newEntry.setGroupId(entry.getGroupId());
240            newEntry.setUserName(entry.getUserName());
241            newEntry.setGroupName(entry.getGroupName());
242            newEntry.setMode(entry.getMode());
243            newEntry.setSize(entry.getSize());
244
245            return newEntry;
246        }
247
248        return pEntry;
249    }
