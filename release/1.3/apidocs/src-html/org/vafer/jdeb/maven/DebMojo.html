340    protected void setData( Data[] dataSet ) {
341        this.dataSet = dataSet;
342        dataProducers.clear();
343        conffileProducers.clear();
344        if (dataSet != null) {
345            Collections.addAll(dataProducers, dataSet);
346            
347            for (Data item : dataSet) {
348                if (item.getConffile()) {
349                    conffileProducers.add(item);
350                }
351            }
352        }
353    }

355    protected VariableResolver initializeVariableResolver( Map<String, String> variables ) {
356        @SuppressWarnings("unchecked")
357        final Map<String, String> projectProperties = Map.class.cast(getProject().getProperties());
358        @SuppressWarnings("unchecked")
359        final Map<String, String> systemProperties = Map.class.cast(System.getProperties());
360
361        variables.putAll(projectProperties);
362        variables.putAll(systemProperties);
363        variables.put("name", name != null ? name : getProject().getName());
364        variables.put("artifactId", getProject().getArtifactId());
365        variables.put("groupId", getProject().getGroupId());
366        variables.put("version", getProjectVersion());
367        variables.put("description", getProject().getDescription());
368        variables.put("extension", "deb");
369        variables.put("baseDir", getProject().getBasedir().getAbsolutePath());
370        variables.put("buildDir", buildDirectory.getAbsolutePath());
371        variables.put("project.version", getProject().getVersion());
372        variables.put("url", getProject().getUrl());
373
374        return new MapVariableResolver(variables);
375    }

422    /**
423     * Main entry point
424     *
425     * @throws MojoExecutionException on error
426     */
427    public void execute() throws MojoExecutionException {
428
429        final MavenProject project = getProject();
430
431        if (skip) {
432            getLog().info("skipping as configured (skip)");
433            return;
434        }
435
436        if (skipPOMs && isPOM()) {
437            getLog().info("skipping because artifact is a pom (skipPOMs)");
438            return;
439        }
440
441        if (skipSubmodules && isSubmodule()) {
442            getLog().info("skipping submodule (skipSubmodules)");
443            return;
444        }
445
446
447        setData(dataSet);
448
449        console = new MojoConsole(getLog(), verbose);
450
451        initializeSignProperties();
452
453        final VariableResolver resolver = initializeVariableResolver(new HashMap<String, String>());
454
455        final File debFile = new File(Utils.replaceVariables(resolver, deb, openReplaceToken, closeReplaceToken));
456        final File controlDirFile = new File(Utils.replaceVariables(resolver, controlDir, openReplaceToken, closeReplaceToken));
457        final File installDirFile = new File(Utils.replaceVariables(resolver, installDir, openReplaceToken, closeReplaceToken));
458        final File changesInFile = new File(Utils.replaceVariables(resolver, changesIn, openReplaceToken, closeReplaceToken));
459        final File changesOutFile = new File(Utils.replaceVariables(resolver, changesOut, openReplaceToken, closeReplaceToken));
460        final File changesSaveFile = new File(Utils.replaceVariables(resolver, changesSave, openReplaceToken, closeReplaceToken));
461        final File keyringFile = keyring == null ? null : new File(Utils.replaceVariables(resolver, keyring, openReplaceToken, closeReplaceToken));
462
463        // if there are no producers defined we try to use the artifacts
464        if (dataProducers.isEmpty()) {
465
466            if (hasMainArtifact()) {
467                Set<Artifact> artifacts = new HashSet<Artifact>();
468
469                artifacts.add(project.getArtifact());
470
471                @SuppressWarnings("unchecked")
472                final Set<Artifact> projectArtifacts = project.getArtifacts();
473
474                for (Artifact artifact : projectArtifacts) {
475                    artifacts.add(artifact);
476                }
477
478                @SuppressWarnings("unchecked")
479                final List<Artifact> attachedArtifacts = project.getAttachedArtifacts();
480
481                for (Artifact artifact : attachedArtifacts) {
482                    artifacts.add(artifact);
483                }
484
485                for (Artifact artifact : artifacts) {
486                    final File file = artifact.getFile();
487                    if (file != null) {
488                        dataProducers.add(new DataProducer() {
489                            @Override
490                            public void produce( final DataConsumer receiver ) {
491                                try {
492                                    receiver.onEachFile(
493                                        new FileInputStream(file),
494                                        new File(installDirFile, file.getName()).getAbsolutePath(),
495                                        "",
496                                        "root", 0, "root", 0,
497                                        TarEntry.DEFAULT_FILE_MODE,
498                                        file.length());
499                                } catch (Exception e) {
500                                    getLog().error(e);
501                                }
502                            }
503                        });
504                    } else {
505                        getLog().error("No file for artifact " + artifact);
506                    }
507                }
508            }
509        }
510
511        try {
512            DebMaker debMaker = new DebMaker(console, dataProducers, conffileProducers);
513            debMaker.setDeb(debFile);
514            debMaker.setControl(controlDirFile);
515            debMaker.setPackage(getProject().getArtifactId());
516            debMaker.setDescription(getProject().getDescription());
517            debMaker.setHomepage(getProject().getUrl());
518            debMaker.setChangesIn(changesInFile);
519            debMaker.setChangesOut(changesOutFile);
520            debMaker.setChangesSave(changesSaveFile);
521            debMaker.setCompression(compression);
522            debMaker.setKeyring(keyringFile);
523            debMaker.setKey(key);
524            debMaker.setPassphrase(passphrase);
525            debMaker.setSignPackage(signPackage);
526            debMaker.setSignMethod(signMethod);
527            debMaker.setSignRole(signRole);
528            debMaker.setResolver(resolver);
529            debMaker.setOpenReplaceToken(openReplaceToken);
530            debMaker.setCloseReplaceToken(closeReplaceToken);
531            debMaker.validate();
532            debMaker.makeDeb();
533
534            // Always attach unless explicitly set to false
535            if ("true".equalsIgnoreCase(attach)) {
536                console.info("Attaching created debian package " + debFile);
537                if (!isType()) {
538                    projectHelper.attachArtifact(project, type, classifier, debFile);
539                } else {
540                    project.getArtifact().setFile(debFile);
541                }
542            }
543
544        } catch (PackagingException e) {
545            getLog().error("Failed to create debian package " + debFile, e);
546            throw new MojoExecutionException("Failed to create debian package " + debFile, e);
547        }
548    }

550    /**
551     * Initializes unspecified sign properties using available defaults
552     * and global settings.
553     */
554    private void initializeSignProperties() {
555        if (!signPackage) {
556            return;
557        }
558
559        if (key != null && keyring != null && passphrase != null) {
560            return;
561        }
562
563        Map<String, String> properties =
564                readPropertiesFromActiveProfiles(signCfgPrefix, KEY, KEYRING, PASSPHRASE);
565
566        key = lookupIfEmpty(key, properties, KEY);
567        keyring = lookupIfEmpty(keyring, properties, KEYRING);
568        passphrase = decrypt(lookupIfEmpty(passphrase, properties, PASSPHRASE));
569
570        if (keyring == null) {
571            try {
572                keyring = Utils.guessKeyRingFile().getAbsolutePath();
573                console.info("Located keyring at " + keyring);
574            } catch (FileNotFoundException e) {
575                console.warn(e.getMessage());
576            }
577        }
578    }

580    /**
581     * Decrypts given passphrase if needed using maven security dispatcher.
582     * See http://maven.apache.org/guides/mini/guide-encryption.html for details.
583     *
584     * @param maybeEncryptedPassphrase possibly encrypted passphrase
585     * @return decrypted passphrase
586     */
587    private String decrypt( final String maybeEncryptedPassphrase ) {
588        if (maybeEncryptedPassphrase == null) {
589            return null;
590        }
591
592        try {
593            final String decrypted = secDispatcher.decrypt(maybeEncryptedPassphrase);
594            if (maybeEncryptedPassphrase.equals(decrypted)) {
595                console.info("Passphrase was not encrypted");
596            } else {
597                console.info("Passphrase was successfully decrypted");
598            }
599            return decrypted;
600        } catch (SecDispatcherException e) {
601            console.warn("Unable to decrypt passphrase: " + e.getMessage());
602        }
603
604        return maybeEncryptedPassphrase;
605    }

621    /**
622     * Read properties from the active profiles.
623     *
624     * Goes through all active profiles (in the order the
625     * profiles are defined in settings.xml) and extracts
626     * the desired properties (if present). The prefix is
627     * used when looking up properties in the profile but
628     * not in the returned map.
629     *
630     * @param prefix The prefix to use or null if no prefix should be used
631     * @param properties The properties to read
632     *
633     * @return A map containing the values for the properties that were found
634     */
635    public Map<String, String> readPropertiesFromActiveProfiles( final String prefix,
636                                                                 final String... properties ) {
637        if (settings == null) {
638            console.debug("No maven setting injected");
639            return Collections.emptyMap();
640        }
641
642        final List<String> activeProfilesList = settings.getActiveProfiles();
643        if (activeProfilesList.isEmpty()) {
644            console.debug("No active profiles found");
645            return Collections.emptyMap();
646        }
647
648        final Map<String, String> map = new HashMap<String, String>();
649        final Set<String> activeProfiles = new HashSet<String>(activeProfilesList);
650
651        // Iterate over all active profiles in order
652        for (final Profile profile : settings.getProfiles()) {
653            // Check if the profile is active
654            final String profileId = profile.getId();
655            if (activeProfiles.contains(profileId)) {
656                console.debug("Trying active profile " + profileId);
657                for (final String property : properties) {
658                    final String propKey = prefix != null ? prefix + property : property;
659                    final String value = profile.getProperties().getProperty(propKey);
660                    if (value != null) {
661                        console.debug("Found property " + property + " in profile " + profileId);
662                        map.put(property, value);
663                    }
664                }
665            }
666        }
667
668        return map;
669    }
