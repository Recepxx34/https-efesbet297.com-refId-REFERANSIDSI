001package org.vafer.jdeb.utils;
002
003import java.io.ByteArrayOutputStream;
004import java.io.IOException;
005import java.io.OutputStream;
006
007import org.bouncycastle.bcpg.ArmoredOutputStream;
008import org.bouncycastle.openpgp.PGPException;
009import org.bouncycastle.openpgp.PGPSignature;
010import org.bouncycastle.openpgp.PGPSignatureGenerator;
011
012/**
013 * An output stream that calculates the signature of the input data as it
014 * is written
015 * 
016 * @author mpoindexter
017 *
018 */
019public class PGPSignatureOutputStream extends OutputStream {
020    private final PGPSignatureGenerator signatureGenerator;
021
022    public PGPSignatureOutputStream( PGPSignatureGenerator signatureGenerator ) {
023        super();
024        this.signatureGenerator = signatureGenerator;
025    }
026
027    public void write( int b ) throws IOException {
028        signatureGenerator.update(new byte[] { (byte)b });
029    }
030
031    public void write( byte[] b ) throws IOException {
032        signatureGenerator.update(b);
033    }
034
035    public void write( byte[] b, int off, int len ) throws IOException {
036        signatureGenerator.update(b, off, len);
037    }
038    
039    public PGPSignature generateSignature() throws PGPException {
040        return signatureGenerator.generate();
041    }
042    
043    public String generateASCIISignature() throws PGPException {
044        try {
045            PGPSignature signature = generateSignature();
046            ByteArrayOutputStream buffer = new ByteArrayOutputStream();
047            ArmoredOutputStream armorStream = new ArmoredOutputStream(buffer);
048            signature.encode(armorStream);
049            armorStream.close();
050            return new String(buffer.toByteArray());
051        } catch(IOException e) {
052            //Should never happen since we are just using a memory buffer
053            throw new RuntimeException(e);
054        }
055    }
056
057}
