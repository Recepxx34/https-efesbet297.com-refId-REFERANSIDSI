46  /**
47   * Builds the control archive of the Debian package.
48   */
49  class ControlBuilder {
50      
51      /** The name of the package maintainer scripts */
52      private static final Set<String> MAINTAINER_SCRIPTS = new HashSet<String>(Arrays.asList("preinst", "postinst", "prerm", "postrm", "config"));
53  
54      /** The name of the other control files subject to token substitution */
55      private static final Set<String> CONFIGURATION_FILENAMES = new HashSet<String>(Arrays.asList("conffiles", "templates", "triggers"));
56  
57      private Console console;
58      private VariableResolver resolver;
59      private final String openReplaceToken;
60      private final String closeReplaceToken;
61  
62      ControlBuilder(Console console, VariableResolver resolver, String openReplaceToken, String closeReplaceToken) {
63          this.console = console;
64          this.resolver = resolver;
65          this.openReplaceToken = openReplaceToken;
66          this.closeReplaceToken = closeReplaceToken;
67      }
68  
69      /**
70       * Build control archive of the deb
71       *
72       * @param packageControlFile the package control file
73       * @param controlFiles the other control information files (maintainer scripts, etc)
74       * @param dataSize  the size of the installed package
75       * @param checksums the md5 checksums of the files in the data archive
76       * @param output
77       * @return
78       * @throws java.io.FileNotFoundException
79       * @throws java.io.IOException
80       * @throws java.text.ParseException
81       */
82      void buildControl(BinaryPackageControlFile packageControlFile, File[] controlFiles, List<String> conffiles, StringBuilder checksums, File output) throws IOException, ParseException {
83          final File dir = output.getParentFile();
84          if (dir != null && (!dir.exists() || !dir.isDirectory())) {
85              throw new IOException("Cannot write control file at '" + output.getAbsolutePath() + "'");
86          }
87  
88          final TarArchiveOutputStream outputStream = new TarArchiveOutputStream(new GZIPOutputStream(new FileOutputStream(output)));
89          outputStream.setLongFileMode(TarArchiveOutputStream.LONGFILE_GNU);
90          
91          boolean foundConffiles = false;
92          
93          // create the final package control file out of the "control" file, copy all other files, ignore the directories
94          for (File file : controlFiles) {
95              if (file.isDirectory()) {
96                  // warn about the misplaced directory, except for directories ignored by default (.svn, cvs, etc)
97                  if (!isDefaultExcludes(file)) {
98                      console.warn("Found directory '" + file + "' in the control directory. Maybe you are pointing to wrong dir?");
99                  }
100                 continue;
101             }
102 
103             if ("conffiles".equals(file.getName())) {
104                 foundConffiles = true;
105             }
106             
107             if (CONFIGURATION_FILENAMES.contains(file.getName()) || MAINTAINER_SCRIPTS.contains(file.getName())) {
108                 FilteredFile configurationFile = new FilteredFile(new FileInputStream(file), resolver);
109                 configurationFile.setOpenToken(openReplaceToken);
110                 configurationFile.setCloseToken(closeReplaceToken);
111                 addControlEntry(file.getName(), configurationFile.toString(), outputStream);
112 
113             } else if (!"control".equals(file.getName())) {
114                 // initialize the information stream to guess the type of the file
115                 InformationInputStream infoStream = new InformationInputStream(new FileInputStream(file));
116                 Utils.copy(infoStream, NullOutputStream.NULL_OUTPUT_STREAM);
117                 infoStream.close();
118 
119                 // fix line endings for shell scripts
120                 InputStream in = new FileInputStream(file);
121                 if (infoStream.isShell() && !infoStream.hasUnixLineEndings()) {
122                     byte[] buf = Utils.toUnixLineEndings(in);
123                     in = new ByteArrayInputStream(buf);
124                 }
125                 
126                 addControlEntry(file.getName(), IOUtils.toString(in), outputStream);
127                 
128                 in.close();
129             }
130         }
131         
132         if ((conffiles != null) && (conffiles.size() > 0)) {
133             if (foundConffiles) {
134                 console.info("Found file 'conffiles' in the control directory. Skipping conffiles generation.");
135             } else {
136                 addControlEntry("conffiles", createPackageConffilesFile(conffiles), outputStream);
137             }
138         } else if ((conffiles != null) && (conffiles.size() == 0)) {
139             console.info("Skipping 'conffiles' generation. No entries provided.");
140         }
141 
142         if (packageControlFile == null) {
143             throw new FileNotFoundException("No 'control' file found in " + controlFiles.toString());
144         }
145         
146         addControlEntry("control", packageControlFile.toString(), outputStream);
147         addControlEntry("md5sums", checksums.toString(), outputStream);
148 
149         outputStream.close();
150     }
151     
152     private String createPackageConffilesFile(final List<String> conffiles) {
153         StringBuilder content = new StringBuilder();
154         
155         if (conffiles != null && !conffiles.isEmpty()) {
156             for (String nextFileName : conffiles) {
157                 content.append(nextFileName).append("\n");
158             }
159         }
160 
161         return content.toString();
162     }

165     /**
166      * Creates a package control file from the specified file and adds the
167      * <tt>Date</tt>, <tt>Distribution</tt> and <tt>Urgency</tt> fields if missing.
168      * The <tt>Installed-Size</tt> field is also initialized to the actual size of
169      * the package. The <tt>Maintainer</tt> field is overridden by the <tt>DEBEMAIL</tt>
170      * and <tt>DEBFULLNAME</tt> environment variables if defined.
171      * 
172      * @param file       the control file
173      * @param pDataSize  the size of the installed package
174      */
175     public BinaryPackageControlFile createPackageControlFile(File file, BigInteger pDataSize) throws IOException, ParseException {
176         FilteredFile controlFile = new FilteredFile(new FileInputStream(file), resolver);
177         BinaryPackageControlFile packageControlFile = new BinaryPackageControlFile(controlFile.toString());
178         
179         if (packageControlFile.get("Distribution") == null) {
180             packageControlFile.set("Distribution", "unknown");
181         }
182 
183         if (packageControlFile.get("Urgency") == null) {
184             packageControlFile.set("Urgency", "low");
185         }
186 
187         packageControlFile.set("Installed-Size", pDataSize.divide(BigInteger.valueOf(1024)).toString());
188 
189         // override the Version if the DEBVERSION environment variable is defined
190         final String debVersion = System.getenv("DEBVERSION");
191         if (debVersion != null) {
192             packageControlFile.set("Version", debVersion);
193             console.debug("Using version'" + debVersion + "' from the environment variables.");
194         }
195 
196 
197         // override the Maintainer field if the DEBFULLNAME and DEBEMAIL environment variables are defined
198         final String debFullName = System.getenv("DEBFULLNAME");
199         final String debEmail = System.getenv("DEBEMAIL");
200 
201         if (debFullName != null && debEmail != null) {
202             final String maintainer = debFullName + " <" + debEmail + ">";
203             packageControlFile.set("Maintainer", maintainer);
204             console.debug("Using maintainer '" + maintainer + "' from the environment variables.");
205         }
206         
207         return packageControlFile;
208     }

211     private static void addControlEntry(final String pName, final String pContent, final TarArchiveOutputStream pOutput) throws IOException {
212         final byte[] data = pContent.getBytes("UTF-8");
213 
214         final TarArchiveEntry entry = new TarArchiveEntry("./" + pName, true);
215         entry.setSize(data.length);
216         entry.setNames("root", "root");
217         
218         if (MAINTAINER_SCRIPTS.contains(pName)) {
219             entry.setMode(PermMapper.toMode("755"));
220         } else {
221             entry.setMode(PermMapper.toMode("644"));
222         }
223         
224         pOutput.putArchiveEntry(entry);
225         pOutput.write(data);
226         pOutput.closeArchiveEntry();
227     }

229     /**
230      * Tells if the specified directory is ignored by default (.svn, cvs, etc)
231      * 
232      * @param directory
233      */
234     private boolean isDefaultExcludes(File directory) {
235         for (String pattern : DirectoryScanner.getDefaultExcludes()) {
236             if (DirectoryScanner.match(pattern, directory.getAbsolutePath().replace("\\", "/"))) {
237                 return true;
238             }
239         }
240         
241         return false;
242     }
