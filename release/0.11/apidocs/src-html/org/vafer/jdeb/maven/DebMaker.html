032    /**
033     * TODO generalize with DebAntTask
034     *
035     * A generic class for creating debian archives. Even supports signed changes
036     * files.
037     *
038     * This class is largely based on the DebAntTask class, and provides the same
039     * featues to maven developers that the ant task provides to ant developers.
040     *
041     * @author Bryan Sant
042     * @see org.vafer.jdeb.ant.DebAntTask
043     */
044    public class DebMaker {
045    
046        /**
047         * A console to output log message with
048         */
049        private Console console;
050    
051        /**
052         * The Debian package produced
053         */
054        private File deb;
055    
056        /**
057         * The directory containing the control files to build the package
058         */
059        private File control;
060    
061        /**
062         * The file containing the PGP keys
063         */
064        private File keyring;
065    
066        /**
067         * The key to use in the keyring
068         */
069        private String key;
070    
071        /**
072         * The passphrase for the key to sign the changes file
073         */
074        private String passphrase;
075    
076        /**
077         * The file to read the changes from
078         */
079        private File changesIn;
080    
081        /**
082         * The file where to write the changes to
083         */
084        private File changesOut;
085    
086        /**
087         * The file where to write the changes of the changes input to
088         */
089        private File changesSave;
090    
091        /**
092         * The compression method used for the data file (none, gzip or bzip2)
093         */
094        private String compression = "gzip";
095    
096        private final VariableResolver variableResolver;
097    
098    
099        private final Collection dataProducers;
100    
101        public DebMaker( Console console, VariableResolver variableResolver ) {
102            this(console, null, null, null, variableResolver);
103        }
104    
105        public DebMaker( Console console, File deb, File controlDir, VariableResolver variableResolver ) {
106            this(console, deb, controlDir, null, variableResolver);
107        }
108    
109        public DebMaker( Console console, File deb, File controlDir, Collection dataProducers, VariableResolver variableResolver ) {
110            this.console = console;
111            this.deb = deb;
112            this.control = controlDir;
113            this.variableResolver = variableResolver;
114            if (dataProducers != null) {
115                this.dataProducers = dataProducers;
116            } else {
117                this.dataProducers = new ArrayList();
118            }
119        }
120    
121        public void setDeb( File deb ) {
122            this.deb = deb;
123        }
124    
125        public void setControl( File control ) {
126            this.control = control;
127        }
128    
129        public void setChangesIn( File changes ) {
130            this.changesIn = changes;
131        }
132    
133        public void setChangesOut( File changes ) {
134            this.changesOut = changes;
135        }
136    
137        public void setChangesSave( File changes ) {
138            this.changesSave = changes;
139        }
140    
141        public void setKeyring( File keyring ) {
142            this.keyring = keyring;
143        }
144    
145        public void setKey( String key ) {
146            this.key = key;
147        }
148    
149        public void setPassphrase( String passphrase ) {
150            this.passphrase = passphrase;
151        }
152    
153        public void setCompression( String compression ) {
154            this.compression = compression;
155        }
156    
157        /**
158         * Adds a new data source to the deb. The new data source may point to a
159         * single file, a directory, or a tarball.
160         *
161         * @param file The data source to add
162         */
163        public void addData( File file ) {
164            final Data data = new Data();
165            data.setSrc(file);
166            dataProducers.add(data);
167        }
168    
169        public void addData( Data data ) {
170            dataProducers.add(data);
171        }
172    
173        private boolean isPossibleOutput( File file ) {
174    
175            if (file.exists()) {
176                return file.isFile() && file.canWrite();
177            }
178    
179            return true;
180        }
181    
182        public void makeDeb() throws PackagingException {
183    
184            if (control == null || !control.isDirectory()) {
185                throw new PackagingException(
186                    "\"" + control + "\" is not a valid 'control' directory)");
187            }
188    
189            if (changesIn != null) {
190    
191                if (!changesIn.isFile() || !changesIn.canRead()) {
192                    throw new PackagingException(
193                        "The 'changesIn' setting needs to point to a readable file. "
194                            + changesIn + " was not found/readable.");
195                }
196    
197                if (changesOut == null) {
198                    throw new PackagingException(
199                        "A 'changesIn' without a 'changesOut' does not make much sense.");
200                }
201    
202                if (!isPossibleOutput(changesOut)) {
203                    throw new PackagingException(
204                        "Cannot write the output for 'changesOut' to "
205                            + changesOut);
206                }
207    
208                if (changesSave != null && !isPossibleOutput(changesSave)) {
209                    throw new PackagingException(
210                        "Cannot write the output for 'changesSave' to "
211                            + changesSave);
212                }
213    
214            } else {
215                if (changesOut != null || changesSave != null) {
216                    throw new PackagingException(
217                        "The 'changesOut' or 'changesSave' settings may only be used when there is a 'changesIn' specified.");
218                }
219            }
220    
221            if (!"gzip".equals(compression) && !"bzip2".equals(compression)
222                && !"none".equals(compression)) {
223                throw new PackagingException("The compression method '"
224                    + compression + "' is not supported");
225            }
226    
227            if (deb == null) {
228                throw new PackagingException(
229                    "You need to specify where the deb file is supposed to be created.");
230            }
231    
232            final File[] controlFiles = control.listFiles();
233    
234            final DataProducer[] data = new DataProducer[dataProducers.size()];
235            dataProducers.toArray(data);
236    
237            final Processor processor = new Processor(console, variableResolver);
238    
239            final PackageDescriptor packageDescriptor;
240            try {
241    
242                console.info("Creating debian package: " + deb);
243    
244                packageDescriptor = processor.createDeb(controlFiles, data, deb, compression);
245    
246            } catch (Exception e) {
247                throw new PackagingException("Failed to create debian package " + deb, e);
248            }
249    
250            final TextfileChangesProvider changesProvider;
251    
252            try {
253                if (changesOut == null) {
254                    return;
255                }
256    
257                console.info("Creating changes file: " + changesOut);
258    
259                // for now only support reading the changes form a textfile provider
260                changesProvider = new TextfileChangesProvider(new FileInputStream(changesIn), packageDescriptor);
261    
262                processor.createChanges(packageDescriptor, changesProvider,
263                    (keyring != null) ? new FileInputStream(keyring) : null,
264                    key, passphrase, new FileOutputStream(changesOut));
265    
266            } catch (Exception e) {
267                throw new PackagingException(
268                    "Failed to create debian changes file " + changesOut, e);
269            }
270    
271            try {
272                if (changesSave == null) {
273                    return;
274                }
275    
276                console.info("Saving changes to file: " + changesSave);
277    
278                changesProvider.save(new FileOutputStream(changesSave));
279    
280            } catch (Exception e) {
281                throw new PackagingException("Failed to save debian changes file " + changesSave, e);
282            }
283    
284        }
