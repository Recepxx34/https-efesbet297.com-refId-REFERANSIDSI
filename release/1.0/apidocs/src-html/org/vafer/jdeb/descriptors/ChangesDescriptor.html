016package org.vafer.jdeb.descriptors;
017
018import org.vafer.jdeb.changes.ChangeSet;
019
020/**
021 * Reflecting a changes file
022 *
023 * @author Torsten Curdt
024 */
025public final class ChangesDescriptor extends AbstractDescriptor {
026
027    private final static String[] keys = {
028        "Format",
029        "Date",
030        "Source",
031        "Binary",
032        "Architecture",
033        "Version",
034        "Distribution",
035        "Urgency",
036        "Maintainer",
037        "Changed-By",
038        "Description",
039        "Changes",
040        "Closes",
041        "Checksums-Sha1",
042        "Checksums-Sha256",
043        "Files"
044    };
045
046    public final static String[] mandatoryKeys = {
047        "Format",
048        "Date",
049        "Source",
050        "Binary",
051        "Architecture",
052        "Version",
053        "Distribution",
054        "Urgency",
055        "Maintainer",
056        "Description",
057        "Changes",
058        "Checksums-Sha1",
059        "Checksums-Sha256",
060        "Files"
061    };
062
063    private final ChangeSet[] changeSets;
064
065    public ChangesDescriptor( final AbstractDescriptor pDescriptor, final ChangeSet[] pChangeSets ) {
066        super(pDescriptor);
067        changeSets = pChangeSets;
068
069        final StringBuilder sb = new StringBuilder();
070
071        if (changeSets.length > 0) {
072            final ChangeSet latestChangeSet = changeSets[0];
073            set("Urgency", latestChangeSet.getUrgency());
074            set("Changed-By", latestChangeSet.getChangedBy());
075
076            for (int i = 0; i < 1; i++) {
077                final ChangeSet changeSet = changeSets[i];
078                sb.append(changeSet.toString());
079            }
080        }
081
082        set("Changes", sb.toString());
083    }
084
085    public String[] getMandatoryKeys() {
086        return mandatoryKeys;
087    }
088
089    public String toString() {
090        return toString(keys);
091    }
092}
