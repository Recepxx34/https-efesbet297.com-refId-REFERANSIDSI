170    /**
171     * "data" entries used to determine which files should be added to this deb.
172     * The "data" entries may specify a tarball (tar.gz, tar.bz2, tgz), a
173     * directory, or a normal file. An entry would look something like this in
174     * your pom.xml:
175     *
176     * <pre>
177     *   <build>
178     *     <plugins>
179     *       <plugin>
180     *       <artifactId>jdeb</artifactId>
181     *       <groupId>org.vafer</groupId>
182     *       ...
183     *       <configuration>
184     *         ...
185     *         <dataSet>
186     *           <data>
187     *             <src>${project.basedir}/target/my_archive.tar.gz</src>
188     *             <include>...</include>
189     *             <exclude>...</exclude>
190     *             <mapper>
191     *               <type>perm</type>
192     *               <strip>1</strip>
193     *               <prefix>/somewhere/else</prefix>
194     *               <user>santbj</user>
195     *               <group>santbj</group>
196     *               <mode>600</mode>
197     *             </mapper>
198     *           </data>
199     *           <data>
200     *             <src>${project.build.directory}/data</src>
201     *             <include></include>
202     *             <exclude>**&#47;.svn</exclude>
203     *             <mapper>
204     *               <type>ls</type>
205     *               <src>mapping.txt</src>
206     *             </mapper>
207     *           </data>
208     *           <data>
209     *             <type>link</type>
210     *             <linkName>/a/path/on/the/target/fs</linkName>
211     *             <linkTarget>/a/sym/link/to/the/scr/file</linkTarget>
212     *             <symlink>true</symlink>
213     *           </data>
214     *           <data>
215     *             <src>${project.basedir}/README.txt</src>
216     *           </data>
217     *         </dataSet>
218     *       </configuration>
219     *     </plugins>
220     *   </build>
221     * </pre>
222     *
223     * @parameter expression="${dataSet}"
224     */
225    private Data[] dataSet;

233    private boolean timestamped;

246    private String openReplaceToken = "[[";
247    private String closeReplaceToken = "]]";
248    private Collection<DataProducer> dataProducers = new ArrayList<DataProducer>();
249
250    public void setOpenReplaceToken( String openReplaceToken ) {
251        this.openReplaceToken = openReplaceToken;
252        // FIXME yuck!
253        AbstractDescriptor.setOpenToken(openReplaceToken);
254    }
255
256    public void setCloseReplaceToken( String closeReplaceToken ) {
257        this.closeReplaceToken = closeReplaceToken;
258        // FIXME yuck!
259        AbstractDescriptor.setCloseToken(closeReplaceToken);
260    }
261
262    protected void setData( Data[] dataSet ) {
263        this.dataSet = dataSet;
264        dataProducers.clear();
265        if (dataSet != null) {
266            Collections.addAll(dataProducers, dataSet);
267        }
268    }
269
270    protected VariableResolver initializeVariableResolver( Map<String, String> variables ) {
271        ((Map) variables).putAll(getProject().getProperties());
272        variables.put("name", name != null ? name : getProject().getName());
273        variables.put("artifactId", getProject().getArtifactId());
274        variables.put("groupId", getProject().getGroupId());
275        variables.put("version", getProjectVersion());
276        variables.put("description", getProject().getDescription());
277        variables.put("extension", "deb");
278        variables.put("baseDir", getProject().getBasedir().getAbsolutePath());
279        variables.put("buildDir", buildDirectory.getAbsolutePath());
280        variables.put("project.version", getProject().getVersion());
281        variables.put("url", getProject().getUrl());
282
283        return new MapVariableResolver(variables);
284    }

286    /**
287     * Doc some cleanup and conversion on the Maven project version.
288     * <ul>
289     * <li>any "-" is replaced by "+"</li>
290     * <li>"SNAPSHOT" is replaced with the current time and date, prepended by "~"</li>
291     * </ul>
292     *
293     * @return the Maven project version
294     */
295    private String getProjectVersion() {
296        return Utils.convertToDebianVersion(getProject().getVersion(), this.timestamped ? session.getStartTime() : null);
297    }
298
299    /**
300     * @return whether or not Maven is currently operating in the execution root
301     */
302    private boolean isSubmodule() {
303        // FIXME there must be a better way
304        return !session.getExecutionRootDirectory().equalsIgnoreCase(baseDir.toString());
305    }
306
307    /**
308     * @return whether or not the main artifact was created
309     */
310    private boolean hasMainArtifact() {
311        final MavenProject project = getProject();
312        final Artifact artifact = project.getArtifact();
313        return artifact.getFile() != null && artifact.getFile().isFile();
314    }
315
316    /**
317     * Main entry point
318     *
319     * @throws MojoExecutionException on error
320     */
321    @Override
322    public void execute() throws MojoExecutionException {
323
324        final MavenProject project = getProject();
325
326        if (isSubmodule() && !submodules) {
327            getLog().info("skipping sub module: jdeb executing at top-level only");
328            return;
329        }
330
331        setData(dataSet);
332
333        Console infoConsole = new MojoConsole(getLog(), verbose);
334
335        final VariableResolver resolver = initializeVariableResolver(new HashMap<String, String>());
336
337        final File debFile = new File(Utils.replaceVariables(resolver, deb, openReplaceToken, closeReplaceToken));
338        final File controlDirFile = new File(Utils.replaceVariables(resolver, controlDir, openReplaceToken, closeReplaceToken));
339        final File installDirFile = new File(Utils.replaceVariables(resolver, installDir, openReplaceToken, closeReplaceToken));
340        final File changesInFile = new File(Utils.replaceVariables(resolver, changesIn, openReplaceToken, closeReplaceToken));
341        final File changesOutFile = new File(Utils.replaceVariables(resolver, changesOut, openReplaceToken, closeReplaceToken));
342        final File changesSaveFile = new File(Utils.replaceVariables(resolver, changesSave, openReplaceToken, closeReplaceToken));
343
344        // if there are no producers defined we try to use the artifacts
345        if (dataProducers.isEmpty()) {
346
347            if (!hasMainArtifact()) {
348
349                final String packaging = project.getPackaging();
350                if ("pom".equalsIgnoreCase(packaging)) {
351                    getLog().warn("Creating empty debian package.");
352                } else {
353                    throw new MojoExecutionException(
354                        "Nothing to include into the debian package. " +
355                            "Did you maybe forget to add a <data> tag or call the plugin directly?");
356                }
357
358            } else {
359
360                Set<Artifact> artifacts = new HashSet<Artifact>();
361
362                artifacts.add(project.getArtifact());
363
364                for (Artifact artifact : (Set<Artifact>) project.getArtifacts()) {
365                    artifacts.add(artifact);
366                }
367
368                for (Artifact artifact : (List<Artifact>) project.getAttachedArtifacts()) {
369                    artifacts.add(artifact);
370                }
371
372                for(Artifact artifact : artifacts) {
373                    final File file = artifact.getFile();
374                    if (file != null) {
375                        dataProducers.add(new DataProducer() {
376                            @Override
377                            public void produce( final DataConsumer receiver ) {
378                                try {
379                                    receiver.onEachFile(
380                                        new FileInputStream(file),
381                                        new File(installDirFile, file.getName()).getAbsolutePath(),
382                                        "",
383                                        "root", 0, "root", 0,
384                                        TarEntry.DEFAULT_FILE_MODE,
385                                        file.length());
386                                } catch (Exception e) {
387                                    getLog().error(e);
388                                }
389                            }
390                        });
391                    } else {
392                        getLog().error("No file for artifact " + artifact);
393                    }
394                }
395            }
396        }
397
398        try {
399
400            DebMaker debMaker = new DebMaker(infoConsole, debFile, controlDirFile, dataProducers, resolver);
401
402            if (changesInFile.exists() && changesInFile.canRead()) {
403                debMaker.setChangesIn(changesInFile);
404                debMaker.setChangesOut(changesOutFile);
405                debMaker.setChangesSave(changesSaveFile);
406            }
407
408            debMaker.setCompression(compression);
409            debMaker.makeDeb();
410
411            // Always attach unless explicitly set to false
412            if ("true".equalsIgnoreCase(attach)) {
413                getLog().info("Attaching created debian archive " + debFile);
414                projectHelper.attachArtifact(project, type, classifier, debFile);
415            }
416
417        } catch (PackagingException e) {
418            getLog().error("Failed to create debian package " + debFile, e);
419            throw new MojoExecutionException("Failed to create debian package " + debFile, e);
420        }
421    }
