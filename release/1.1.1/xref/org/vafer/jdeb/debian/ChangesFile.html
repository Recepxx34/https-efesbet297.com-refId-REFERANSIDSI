17  package org.vafer.jdeb.debian;
18  
19  import org.vafer.jdeb.changes.ChangeSet;
20  
21  import java.text.DateFormat;
22  import java.text.SimpleDateFormat;
23  import java.util.Date;
24  import java.util.Locale;
25  import java.util.Map.Entry;
26  
27  /**
28   * Reflecting a changes file
29   *
30   * @see <a href="http://www.debian.org/doc/debian-policy/ch-controlfields.html#s-debianchangesfiles">Debian Policy Manual - Debian changes files</a>
31   * @author Torsten Curdt
32   */
33  public final class ChangesFile extends ControlFile {
34  
35      private static final ControlField[] FIELDS = {
36              new ControlField("Format", true),
37              new ControlField("Date", true),
38              new ControlField("Source", true),
39              new ControlField("Binary", true),
40              new ControlField("Architecture", true),
41              new ControlField("Version", true),
42              new ControlField("Distribution", true),
43              new ControlField("Urgency", true),
44              new ControlField("Maintainer", true),
45              new ControlField("Changed-By"),
46              new ControlField("Description", true, ControlField.Type.MULTILINE, true),
47              new ControlField("Changes", true, ControlField.Type.MULTILINE, true),
48              new ControlField("Closes"),
49              new ControlField("Checksums-Sha1", true, ControlField.Type.MULTILINE, true),
50              new ControlField("Checksums-Sha256", true, ControlField.Type.MULTILINE, true),
51              new ControlField("Files", true, ControlField.Type.MULTILINE, true)
52      };
53  
54      public ChangesFile() {
55          set("Format", "1.8");
56          set("Urgency", "low");
57          set("Distribution", "stable");
58      }
59  
60      /**
61       * Initializes the fields on the changes file with the values of the specified
62       * binary package control file.
63       *
64       * @param packageControlFile
65       */
66      public void initialize(BinaryPackageControlFile packageControlFile) {
67          set("Binary",       packageControlFile.get("Package"));
68          set("Source",       packageControlFile.get("Package"));
69          set("Architecture", packageControlFile.get("Architecture"));
70          set("Version",      packageControlFile.get("Version"));
71          set("Maintainer",   packageControlFile.get("Maintainer"));
72          set("Changed-By",   packageControlFile.get("Maintainer"));
73          set("Distribution", packageControlFile.get("Distribution"));
74  
75          for (Entry<String, String> entry : packageControlFile.getUserDefinedFields().entrySet()) {
76              set(entry.getKey(), entry.getValue());
77          }
78  
79          StringBuilder description = new StringBuilder();
80          description.append(packageControlFile.get("Package"));
81          if (packageControlFile.get("Description") != null) {
82              description.append(" - ");
83              description.append(packageControlFile.getShortDescription());
84          }
85          set("Description",  description.toString());
86      }
87  
88      public void setChanges(ChangeSet[] changeSets) {
89          StringBuilder sb = new StringBuilder();
90  
91          if (changeSets.length > 0) {
92              final ChangeSet mostRecentChangeSet = changeSets[0];
93              set("Urgency", mostRecentChangeSet.getUrgency());
94              set("Changed-By", mostRecentChangeSet.getChangedBy());
95  
96              for (ChangeSet changeSet : changeSets) {
97                  sb.append(changeSet.toString());
98              }
99          }
100 
101         set("Changes", sb.toString());
102     }
103 
104     @Override
105     protected ControlField[] getFields() {
106         return FIELDS;
107     }
108 
109     @Override
110     protected char getUserDefinedFieldLetter() {
111         return 'C';
112     }
113 
114     public static String formatDate(Date date) {
115         final DateFormat format = new SimpleDateFormat("EEE, d MMM yyyy HH:mm:ss Z", Locale.ENGLISH); // RFC 2822 format
116         return format.format(date);
117     }
